016    package com.github.gwtbootstrap.client.ui;
017    
018    import com.github.gwtbootstrap.client.ui.resources.Bootstrap;
019    import com.octo.gwt.test.GwtTest;
020    import org.junit.Test;
021    
022    import static org.junit.Assert.*;
023    
024    
025    /**
026     * This is {@link Abbreviation}Test.
027     * @author ohashi keisuke
028     */
029    public class AbbreviationTest extends GwtTest {
030    
031            @Override
032            public String getModuleName() {
033                    return "com.github.gwtbootstrap.Bootstrap";
034            }
035    
036            /**
037             * test for Constractor {@link Abbreviation#Abbreviation(String, String)}
038             */
039            @Test
040            public void testAbbreviation() {
041                    
042                    //given
043                    String text = "text";
044                    String title = "title";
045                    
046                    //actual
047                    Abbreviation abbreviation = new Abbreviation(text, title);
048                    
049                    //assert
050                    assertNotNull(abbreviation);
051    
052                    assertEquals("tag should be addr","abbr", abbreviation.getElement().getTagName());
053                    
054                    assertEquals("text should be args",text , abbreviation.getText());
055                    
056                    assertEquals("title should be args" ,title, abbreviation.getTitle());
057            }
058    
059            /**
060             * test for Constractor {@link Abbreviation#Abbreviation(String)}
061             */
062            @Test
063            public void testAbbreviationString() {
064                    //given
065                    String title = "title";
066                    
067                    //actual
068                    Abbreviation abbreviation = new Abbreviation(title);
069                    
070                    //assert
071                    assertNotNull(abbreviation);
072    
073                    assertEquals("tag should be addr","abbr", abbreviation.getElement().getTagName());
074                    
075                    assertEquals("text should be empty (not null)","" , abbreviation.getText());
076                    
077                    assertEquals("title should be args" ,title, abbreviation.getTitle());
078            }
079    
080            /**
081             * test for {@link Abbreviation#setInitalism(boolean)}method
082             */
083            @Test
084            public void testSetInitalism() {
085                    //given
086                    String title = "title";
087                    Abbreviation abbreviation = new Abbreviation(title);
088                    
089                    //actual
090                    abbreviation.setInitalism(true);
091                    
092                    String styleName = abbreviation.getStyleName();
093                    
094                    //assert
095                    assertNotNull(styleName);
096                    
097                    assertTrue("if set true,add stylename",styleName.contains(Bootstrap.initialism));
098                    
099                    //actual 2
100                    abbreviation.setInitalism(false);
101                    
102                    styleName = abbreviation.getStyleName();
103                    
104                    //assert
105                    assertNotNull(styleName);
106                    assertFalse("if set false,remove stylename",styleName.contains(Bootstrap.initialism));
107                    
108                    
109            }
110    
111    
112    }
