001    package com.github.gwtbootstrap.client.ui;
002    
003    import static org.junit.Assert.*;
004    
005    import org.junit.Test;
006    
007    import com.github.gwtbootstrap.client.ui.resources.Bootstrap;
008    import com.octo.gwt.test.GwtTest;
009    
010    
011    /**
012     * This is {@link Abbreviation}Test.
013     * @author ohashi keisuke
014     */
015    public class AbbreviationTest extends GwtTest {
016    
017            @Override
018            public String getModuleName() {
019                    return "com.github.gwtbootstrap.Bootstrap";
020            }
021    
022            /**
023             * test for Constractor {@link Abbreviation#Abbreviation(String, String)}
024             */
025            @Test
026            public void testAbbreviation() {
027                    
028                    //given
029                    String text = "text";
030                    String title = "title";
031                    
032                    //actual
033                    Abbreviation abbreviation = new Abbreviation(text, title);
034                    
035                    //assert
036                    assertNotNull(abbreviation);
037    
038                    assertEquals("tag should be addr","abbr", abbreviation.getElement().getTagName());
039                    
040                    assertEquals("text should be args",text , abbreviation.getText());
041                    
042                    assertEquals("title should be args" ,title, abbreviation.getTitle());
043            }
044    
045            /**
046             * test for Constractor {@link Abbreviation#Abbreviation(String)}
047             */
048            @Test
049            public void testAbbreviationString() {
050                    //given
051                    String title = "title";
052                    
053                    //actual
054                    Abbreviation abbreviation = new Abbreviation(title);
055                    
056                    //assert
057                    assertNotNull(abbreviation);
058    
059                    assertEquals("tag should be addr","abbr", abbreviation.getElement().getTagName());
060                    
061                    assertEquals("text should be empty (not null)","" , abbreviation.getText());
062                    
063                    assertEquals("title should be args" ,title, abbreviation.getTitle());
064            }
065    
066            /**
067             * test for {@link Abbreviation#setInitalism(boolean)}method
068             */
069            @Test
070            public void testSetInitalism() {
071                    //given
072                    String title = "title";
073                    Abbreviation abbreviation = new Abbreviation(title);
074                    
075                    //actual
076                    abbreviation.setInitalism(true);
077                    
078                    String styleName = abbreviation.getStyleName();
079                    
080                    //assert
081                    assertNotNull(styleName);
082                    
083                    assertTrue("if set true,add stylename",styleName.contains(Bootstrap.initialism));
084                    
085                    //actual 2
086                    abbreviation.setInitalism(false);
087                    
088                    styleName = abbreviation.getStyleName();
089                    
090                    //assert
091                    assertNotNull(styleName);
092                    assertFalse("if set false,remove stylename",styleName.contains(Bootstrap.initialism));
093                    
094                    
095            }
096    
097    
098    }
