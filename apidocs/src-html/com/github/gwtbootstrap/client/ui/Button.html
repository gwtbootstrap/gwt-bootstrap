090    /**
091     * Button with optional icon.
092     * 
093     * <p>
094     * <h3>UiBinder Usage:</h3>
095     * 
096     * <pre>
097     * {@code <b:Button icon="TRASH" type="ERROR">Delete</b:Button>}
098     * </pre>
099     * 
100     * All arguments are optional.
101     * </p>
102     * 
103     * @since 2.0.2.0
104     * 
105     * @author Carlos Alexandro Becker
106     * 
107     * @author Dominik Mayer
108     * 
109     * @see <a href="http://twitter.github.com/bootstrap/base-css.html#buttons">Bootstrap documentation</a>
110     * @see ButtonGroup
111     * @see ButtonToolbar
112     * @see DropdownButton
113     * @see SplitDropdownButton
114     * @see NavbarButton
115     */
116    //@formatter:on
117    public class Button extends IconAnchor implements HasClickHandlers,
118                    HasDoubleClickHandlers, HasEnabled, HasType<ButtonType>,
119                    HasAllDragAndDropHandlers, HasAllFocusHandlers, HasAllGestureHandlers,
120                    HasAllKeyHandlers, HasAllMouseHandlers, HasAllTouchHandlers {
121    
122            /**
123             * Creates an empty Button.
124             */
125            public Button() {
126                    super();
127                    addStyleName(Constants.BTN);
128            }
129    
130            /**
131             * Creates a Button with the given caption.
132             * 
133             * @param caption
134             *            the caption of the Button
135             */
136            public Button(String caption) {
137                    this();
138                    setText(caption);
139            }
140    
141            /**
142             * Creates a Button with the given caption and icon.
143             * 
144             * @param caption
145             *            the caption of the Button
146             * @param icon
147             *            the Icon on the caption's left
148             */
149            public Button(String caption, IconType icon) {
150                    this(caption);
151                    setIcon(icon);
152            }
153    
154            /**
155             * Sets the type of the Button.
156             * <p>
157             * Different types give the button a different look.
158             * 
159             * @param type
160             *            the type of the Button.
161             */
162            public void setType(ButtonType type) {
163                    for (ButtonType t : ButtonType.values())
164                            removeStyle(t);
165                    addStyle(type);
166            }
167    
168            /**
169             * Sets the size of the Button.
170             * 
171             * @param size
172             *            the size of the Button.
173             */
174            public void setSize(ButtonSize size) {
175                    for (ButtonSize s : ButtonSize.values())
176                            removeStyle(s);
177                    addStyle(size);
178            }
179    
180            /**
181             * Whether the Button is enabled or disabled.
182             * <p>
183             * A disabled widget cannot be used.
184             * 
185             * @return <code>false</code> if the Button is disabled.
186             */
187            public boolean isEnabled() {
188                    return getStyleName().contains(Constants.DISABLED);
189            }
190    
191            /**
192             * Sets whether the Button is enabled or disabled.
193             * <p>
194             * A disabled widget cannot be used.
195             * 
196             * @param enabled
197             *            <code>false</code> if the Button should be disabled. Default:
198             *            <code>true</code>
199             */
200            public void setEnabled(boolean enabled) {
201                    if (enabled)
202                            removeStyleName(Constants.DISABLED);
203                    else
204                            addStyleName(Constants.DISABLED);
205            }
206

212            }
