091/**
092 * Button with optional icon.
093 * 
094 * <p>
095 * <h3>UiBinder Usage:</h3>
096 * 
097 * <pre>
098 * {@code <b:Button icon="TRASH" type="ERROR">Delete</b:Button>}
099 * </pre>
100 * 
101 * All arguments are optional.
102 * </p>
103 * 
104 * @since 2.0.2.0
105 * 
106 * @author Carlos Alexandro Becker
107 * 
108 * @author Dominik Mayer
109 * 
110 * @see <a href="http://twitter.github.com/bootstrap/base-css.html#buttons">Bootstrap documentation</a>
111 * @see ButtonGroup
112 * @see ButtonToolbar
113 * @see DropdownButton
114 * @see SplitDropdownButton
115 * @see NavbarButton
116 */
117//@formatter:on
118public class Button extends IconAnchor implements HasClickHandlers,
119                HasDoubleClickHandlers, HasEnabled, HasType<ButtonType>,
120                HasAllDragAndDropHandlers, HasAllFocusHandlers, HasAllGestureHandlers,
121                HasAllKeyHandlers, HasAllMouseHandlers, HasAllTouchHandlers {
122
123        /**
124         * Creates an empty Button.
125         */
126        public Button() {
127                super();
128                addStyleName(Constants.BTN);
129        }
130
131        /**
132         * Creates a Button with the given caption.
133         * 
134         * @param caption
135         *            the caption of the Button
136         */
137        public Button(String caption) {
138                this();
139                setText(caption);
140        }
141
142        /**
143         * Creates a Button with the given caption and icon.
144         * 
145         * @param caption
146         *            the caption of the Button
147         * @param icon
148         *            the Icon on the caption's left
149         */
150        public Button(String caption, IconType icon) {
151                this(caption);
152                setIcon(icon);
153        }
154
155        /**
156         * Sets the type of the Button.
157         * <p>
158         * Different types give the button a different look.
159         * 
160         * @param type
161         *            the type of the Button.
162         */
163        public void setType(ButtonType type) {
164                for (ButtonType t : ButtonType.values())
165                        removeStyle(t);
166                addStyle(type);
167        }
168
169        /**
170         * Sets the size of the Button.
171         * 
172         * @param size
173         *            the size of the Button.
174         */
175        public void setSize(ButtonSize size) {
176                for (ButtonSize s : ButtonSize.values())
177                        removeStyle(s);
178                addStyle(size);
179        }
180
181        /**
182         * Whether the Button is enabled or disabled.
183         * <p>
184         * A disabled widget cannot be used.
185         * 
186         * @return <code>false</code> if the Button is disabled.
187         */
188        public boolean isEnabled() {
189                return getStyleName().contains(Constants.DISABLED);
190        }
191
192        /**
193         * Sets whether the Button is enabled or disabled.
194         * <p>
195         * A disabled widget cannot be used.
196         * 
197         * @param enabled
198         *            <code>false</code> if the Button should be disabled. Default:
199         *            <code>true</code>
200         */
201        public void setEnabled(boolean enabled) {
202                if (enabled)
203                        removeStyleName(Constants.DISABLED);
204                else
205                        addStyleName(Constants.DISABLED);
206        }

213        }
