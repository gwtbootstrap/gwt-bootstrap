008public class NameValuePairImpl implements NameValuePair {
009    private String name;
010    private String value;
011
012    public NameValuePairImpl(String name, String value) {
013        this.name = name;
014        this.value = value;
015    }
016
017    @Override
018    public String name() {
019        return name;
020    }
021
022    @Override
023    public String value() {
024        return value;
025    }
026}
