016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.base.AddOn;
019import com.github.gwtbootstrap.client.ui.base.DivWidget;
020import com.github.gwtbootstrap.client.ui.constants.IconType;
021
022
023/**
024 * append/prepend {@link AddOn} for input tags.
025 * <p>
026 * If you want to append/prepend add-on,<br/>
027 * Set appendText/prependText or appendIcon/prependIcon attribute.
028 * 
029 * <h3>UiBinder Usage:</h3>
030 * <pre>
031 * {@code
032 * <!-- append text -->
033 * <b:InputAddOn appendText="@">
034 *   <g:TextBox/>
035 * </b:InputAddOn>
036 * 
037 * <!-- above setting is same as blow -->
038 * <div class="input-append">
039 *   <span class="add-on">@</span><input type="text">
040 * </div>
041 * 
042 * 
043 * <!-- prepend text-->
044 * <b:InputAddOn prependText="@">
045 *   <g:TextBox/>
046 * </b:InputAddOn>
047 * 
048 * <!-- above setting is same as blow -->
049 * <div class="input-prepend">
050 *   <input type="text"><span class="add-on">@</span>
051 * </div>
052 * 
053 * <!-- append icon and prepend text-->
054 * <b:InputAddOn appendIcon="STAR" prependText="@">
055 *   <g:TextBox/>
056 * </b:InputAddOn>
057 * 
058 * <!-- above setting is same as blow -->
059 * <div class="input-append input-prepend">
060 *   <span class="add-on"><i class="icon-star"></i></span><input type="text"><span class="add-on">@</span>
061 * </div>
062 * }
063 * </pre>
064 * </p>
065 * 
066 * @since 2.0.3.0
067 * 
068 * @author ohashi keisuke
069 * @see AddOn
070 * @see http://twitter.github.com/bootstrap/base-css.html#forms
071 */
072public class InputAddOn extends DivWidget {
073
074        /** prepend add-on */
075        private AddOn prependAddOn = new AddOn();
076        
077        /** append-add-on */
078        private AddOn appendAddOn = new AddOn();
079        
080        /** set append add-on */
081        private boolean append = false;
082        
083        /**
084         * Creates an empty widget.
085         */
086        public InputAddOn() {
087                super();
088        }
089        
090        /**
091         * set prepend text
092         * @param prependText prepend text
093         */
094        public void setPrependText(String prependText) {
095                setPrependStyle();
096                
097                prependAddOn.setText(prependText);
098        }
099
100        /**
101         * set prepend icon
102         * @param prependIcon IconType
103         */
104        public void setPrependIcon(IconType prependIcon) {
105                setPrependStyle();
106                
107                prependAddOn.setIcon(prependIcon);
108        }
109
110        private void setPrependStyle() {
111                this.setStyleName("input-prepend", true);
112                if(!prependAddOn.isAttached()) {
113                        insert(prependAddOn, 0);
114                }
115        }
116        
117        /**
118         * set append text
119         * @param appendText append text
120         */
121        public void setAppendText(String appendText) {
122                setAppendStyle();
123                
124                appendAddOn.setText(appendText);
125        }
126        
127        /**
128         * set append icon
129         * @param appendIcon append icon
130         */
131        public void setAppendIcon(IconType appendIcon) {
132                setAppendStyle();
133                
134                appendAddOn.setIcon(appendIcon);
135        }
136
137        private void setAppendStyle() {
138                this.setStyleName("input-append" , true);
139                append = true;
140        }
141        
142        /**
143         * {@inheritDoc}
144         */
145        @Override
146        protected void onLoad() {
147                
148                //if setup append add-on and it is not attached,add as last child.
149                if(append && !appendAddOn.isAttached()) {
150                        add(appendAddOn);
151                }
152                
153                super.onLoad();
154        }
155        
156}
