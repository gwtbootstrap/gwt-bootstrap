001    package com.github.gwtbootstrap.client.ui;
002    
003    import com.github.gwtbootstrap.client.ui.base.ComplexWidget;
004    import com.github.gwtbootstrap.client.ui.base.HasIcon;
005    import com.github.gwtbootstrap.client.ui.constants.IconSize;
006    import com.github.gwtbootstrap.client.ui.constants.IconType;
007    import com.google.gwt.user.client.ui.HasText;
008    
009    
010    /**
011     * input add-on.
012     * <p>
013     * AddOn can set text or icon.
014     * <h3>UiBinder Usage:</h3>
015     * 
016     * <pre>
017     * {@code
018     * <!-- text addon -->
019     * <b:AddOn text="@"/>
020     * 
021     * <!-- text addon another usage -->
022     * <b:AddOn>@</b:AddOn>
023     * 
024     * <!-- icon addon -->
025     * <b:AddOn icon="STAR"/>
026     * }
027     * </pre>
028     * </p>
029     * 
030     * @since 2.0.3.0
031     * 
032     * @author ohashi keisuke
033     * @see InputAddOn
034     * @see IconType
035     * @see http://twitter.github.com/bootstrap/base-css.html#forms
036     */
037    public class AddOn extends ComplexWidget implements HasText, HasIcon {
038    
039            /** text */
040            private String text;
041            
042            /** icon */
043            private Icon icon = new Icon();
044    
045            /**
046             * Creates an empty widget.
047             */
048            public AddOn() {
049                    super("span");
050                    setStyleName("add-on");
051            }
052            
053            /**
054             * {@inheritDoc}
055             */
056            public String getText() {
057                    return getElement().getInnerText();
058            }
059            
060            /**
061             * {@inheritDoc}
062             */
063            public void setText(String text) {
064                    this.text = text;
065    
066                    if(icon.isAttached()) {
067                            icon.removeFromParent();
068                    }
069                    getElement().setInnerText(text);
070            }
071            
072            /**
073             * {@inheritDoc}
074             */
075            public void setIcon(IconType iconType) {
076                    
077                    if(text != null) {
078                            getElement().setInnerHTML("");
079                    }
080    
081                    icon.setType(iconType);
082                    
083                    if(!icon.isAttached()) {
084                            add(icon);
085                    }
086            }
087    
088        @Override
089        public void setIconSize(IconSize size) {
090            icon.setIconsSize(size);
091        }
092    
093    }
