005    /**
006     * Interface for has Placeholder attribute element.
007     * @author ohashi keisuke
008     * @see PlaceholderHelper
009     */
010    public interface HasPlaceholder {
011    
012            /**
013             * Set a placeholder attribute (HTML5).
014             * 
015             * @param placeholder
016             *            the String to show. Use a empty string or null to remove the attribute.
017             */
018            public void setPlaceholder(String placeholder);
019            
020            /**
021             * Get a placeholder attribute (HTML5).
022             * 
023             * @return placeholder attribute.
024             */
025            public String getPlaceholder();
026            
027            
028    }
