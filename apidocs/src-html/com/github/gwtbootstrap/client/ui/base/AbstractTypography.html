016    package com.github.gwtbootstrap.client.ui.base;
017    
018    import com.github.gwtbootstrap.client.ui.constants.Device;
019    import com.github.gwtbootstrap.client.ui.constants.ResponsiveStyle;
020    import com.google.gwt.user.client.DOM;
021    import com.google.gwt.user.client.ui.HasText;
022    import com.google.gwt.user.client.ui.Widget;
023    
024    /**
025     * Helper class for widgets that have text.
026     * 
027     * @since 2.0.2.0
028     * 
029     * @author Carlos Alexandro Becker
030     * 
031     */
032    public abstract class AbstractTypography extends Widget implements HasText,
033                    HasStyle, IsResponsive {
034    
035            /**
036             * Creates a new widget without setting a DOM element.
037             */
038            public AbstractTypography() {
039                    super();
040            }
041    
042            /**
043             * Creates a new widget based on the given HTML tag.
044             * 
045             * @param tag
046             *            the HTML tag to be used
047             */
048            public AbstractTypography(String tag) {
049                    setElement(DOM.createElement(tag));
050            }
051    
052            /**
053             * {@inheritDoc}
054             */
055            public void setText(String text) {
056                    getElement().setInnerText(text);
057            }
058    
059            /**
060             * {@inheritDoc}
061             */
062            public String getText() {
063                    return getElement().getInnerText();
064            }
065    
066            /**
067             * {@inheritDoc}
068             */
069            public void setStyle(Style style) {
070                    setStyleName(style.get());
071            }
072    
073            /**
074             * {@inheritDoc}
075             */
076            public void addStyle(Style style) {
077                    addStyleName(style.get());
078            }
079    
080            /**
081             * {@inheritDoc}
082             */
083            public void removeStyle(Style style) {
084                    String styleString = style.get();
085                    if (!styleString.isEmpty())
086                            removeStyleName(styleString);
087            }
088    
089            /**
090             * {@inheritDoc}
091             */
092            public void setShowOn(Device device) {
093                    removeResponsiveStyles();
094                    switch (device) {
095                            case PHONE:
096                                    addStyle(ResponsiveStyle.VISIBLE_PHONE);
097                                    break;
098                            case TABLET:
099                                    addStyle(ResponsiveStyle.VISIBLE_TABLET);
100                                    break;
101                            case DESKTOP:
102                                    addStyle(ResponsiveStyle.VISIBLE_DESKTOP);
103                                    break;
104                    }
105            }
106    
107            private void removeResponsiveStyles() {
108                    removeStyle(ResponsiveStyle.VISIBLE_PHONE);
109                    removeStyle(ResponsiveStyle.VISIBLE_TABLET);
110                    removeStyle(ResponsiveStyle.VISIBLE_DESKTOP);
111                    removeStyle(ResponsiveStyle.HIDDEN_PHONE);
112                    removeStyle(ResponsiveStyle.HIDDEN_TABLET);
113                    removeStyle(ResponsiveStyle.HIDDEN_DESKTOP);
114            }
115    
116            /**
117             * {@inheritDoc}
118             */
119            public void setHideOn(Device device) {
120                    removeResponsiveStyles();
121                    switch (device) {
122                            case PHONE:
123                                    addStyle(ResponsiveStyle.HIDDEN_PHONE);
124                                    break;
125                            case TABLET:
126                                    addStyle(ResponsiveStyle.HIDDEN_TABLET);
127                                    break;
128                            case DESKTOP:
129                                    addStyle(ResponsiveStyle.HIDDEN_DESKTOP);
130                                    break;
131                    }
132            }
133    }
