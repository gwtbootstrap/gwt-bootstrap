016package com.github.gwtbootstrap.client.ui.base;
017
018import com.github.gwtbootstrap.client.ui.constants.Placement;
019import com.github.gwtbootstrap.client.ui.constants.Trigger;
020import com.github.gwtbootstrap.client.ui.constants.VisibilityChange;
021import com.google.gwt.dom.client.Element;
022import com.google.gwt.event.logical.shared.AttachEvent;
023import com.google.gwt.user.client.ui.HasOneWidget;
024import com.google.gwt.user.client.ui.HasText;
025import com.google.gwt.user.client.ui.HasWidgets;
026import com.google.gwt.user.client.ui.IsWidget;
027import com.google.gwt.user.client.ui.Widget;
028
029//@formatter:off
030/**
031* Base class for widgets that hover above other widgets.
032* 
033* @since 2.0.4.0
034* 
035* @author Dominik Mayer
036* 
037* @see <a href="http://twitter.github.com/bootstrap/javascript.html#popovers">Bootstrap documentation</a>
038*/
039//@formatter:on
040public abstract class HoverBase extends MarkupWidget  implements IsWidget, HasWidgets, HasOneWidget, IsAnimated, HasTrigger, HasPlacement, HasText, HasShowDelay, HasVisibility {
041
042        /**
043         * Whether the widget is animated or not.
044         */
045        protected boolean animated = true;
046
047        /**
048         * The placement of the widget relative to its trigger element.
049         */
050        protected Placement placement = Placement.TOP;
051
052        /**
053         * The action that triggers the widget.
054         */
055        protected Trigger trigger = Trigger.HOVER;
056
057        /**
058         * The delay until the widget is shown.
059         */
060        protected int showDelayInMilliseconds = 0;
061
062        /**
063         * The delay until the widget is hidden.
064         */
065        protected int hideDelayInMilliseconds = 0;
066
067        /**
068         * Creates a new widget based on the provided HTML tag.
069         */
070        public HoverBase() {
071        }
072
073        /**
074         * {@inheritDoc}
075         */
076        @Override
077        public Widget asWidget() {
078                
079            if(getWidget() != null) {
080                removeDataIfExists();
081                
082                reconfigure();
083                
084                    getWidget().addAttachHandler(new AttachEvent.Handler() {
085                        
086                        @Override
087                        public void onAttachOrDetach(AttachEvent event) {
088                            if (!event.isAttached()) {
089                                changeVisibility(VisibilityChange.HIDE);
090                            }
091                        }
092                    });
093                }
094                
095                return getWidget();
096
097        }
098        
099        protected void removeDataIfExists() {
100                removeDataIfExists(getWidget().getElement(), getDataName());
101        }
102        
103        protected native void removeDataIfExists(Element e, String dataName) /*-{
104                if($wnd.jQuery(e).data(dataName)) {
105                        var data = $wnd.jQuery(e).data(dataName);
106                        var eventIn, eventOut;
107                        if (data.options.trigger != 'manual') {
108                                eventIn  = data.options.trigger == 'hover' ? 'mouseenter' : 'focus'
109                                eventOut = data.options.trigger == 'hover' ? 'mouseleave' : 'blur'
110                                data.$element.off(eventIn);
111                                data.$element.off(eventOut);
112                        }
113                        $wnd.jQuery(e).removeData(dataName);
114                }
115        }-*/;
116
117        /**
118         * Adds an HTML data attribute to the widget's tag.
119         * 
120         * @param e target element
121         * 
122         * @param attribute
123         *            the name of the attribute without leading <code>"data-"</code>
124         * @param value
125         *            the value to be stored
126         */
127        protected void setDataAttribute(Element e , String attribute, String value) {
128                e.setAttribute("data-" + attribute, value);
129        }
130
131        /**
132         * {@inheritDoc}
133         */
134        public void setAnimation(boolean animated) {
135                this.animated = animated;
136        }
137
138        /**
139         * Redraws the widget with the currently set options. This must <b>not</b>
140         * be called when a parameter is updated because it would deactivate all
141         * other parameters. No idea why...
142         */
143        public abstract void reconfigure();
144
145        /**
146         * {@inheritDoc}
147         */
148        public boolean getAnimation() {
149                return animated;
150        }
151
152        /**
153         * {@inheritDoc} Relative to its trigger element.
154         */
155        public void setPlacement(Placement placement) {
156                this.placement = placement;
157        }
158
159        /**
160         * {@inheritDoc}
161         */
162        public Placement getPlacement() {
163                return placement;
164        }
165
166        /**
167         * {@inheritDoc}
168         */
169        public void setTrigger(Trigger trigger) {
170                this.trigger = trigger;
171        }
172
173        /**
174         * {@inheritDoc}
175         */
176        public Trigger getTrigger() {
177                return trigger;
178        }
179
180        /**
181         * {@inheritDoc}
182         */
183        public void setShowDelay(int delayInMilliseconds) {
184                showDelayInMilliseconds = delayInMilliseconds;
185        }
186
187        /**
188         * {@inheritDoc}
189         */
190        public int getShowDelay() {
191                return showDelayInMilliseconds;
192        }
193
194        /**
195         * {@inheritDoc}
196         */
197        public void setHideDelay(int delayInMilliseconds) {
198                hideDelayInMilliseconds = delayInMilliseconds;
199        }
200
201        /**
202         * {@inheritDoc}
203         */
204        public int getHideDelay() {
205                return hideDelayInMilliseconds;
206        }
207
208        /**
209         * {@inheritDoc}
210         */
211        public void show() {
212                changeVisibility(VisibilityChange.SHOW);
213        }
214
215        /**
216         * {@inheritDoc}
217         */
218        public void hide() {
219                changeVisibility(VisibilityChange.HIDE);
220        }
221
222        /**
223         * {@inheritDoc}
224         */
225        public void toggle() {
226                changeVisibility(VisibilityChange.TOGGLE);
227        }
228
229        /**
230         * Changes the visibility of the widget.
231         * 
232         * @param visibilityChange
233         *            the action to be performed
234         */
235        protected abstract void changeVisibility(VisibilityChange visibilityChange);
236
237        /**
238         * Get data name of JS Data API.
239         * @return data name
240         */
241        protected abstract String getDataName();
242}
