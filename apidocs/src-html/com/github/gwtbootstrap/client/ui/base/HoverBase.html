016package com.github.gwtbootstrap.client.ui.base;
017
018import java.util.Iterator;
019import java.util.NoSuchElementException;
020
021import com.github.gwtbootstrap.client.ui.constants.Placement;
022import com.github.gwtbootstrap.client.ui.constants.Trigger;
023import com.github.gwtbootstrap.client.ui.constants.VisibilityChange;
024import com.google.gwt.dom.client.Element;
025import com.google.gwt.event.logical.shared.AttachEvent;
026import com.google.gwt.user.client.ui.HasOneWidget;
027import com.google.gwt.user.client.ui.HasText;
028import com.google.gwt.user.client.ui.HasWidgets;
029import com.google.gwt.user.client.ui.IsWidget;
030import com.google.gwt.user.client.ui.Widget;
031
032//@formatter:off
033/**
034* Base class for widgets that hover above other widgets.
035* 
036* @since 2.0.4.0
037* 
038* @author Dominik Mayer
039* 
040* @see <a href="http://twitter.github.com/bootstrap/javascript.html#popovers">Bootstrap documentation</a>
041*/
042//@formatter:on
043public abstract class HoverBase implements IsWidget, HasWidgets, HasOneWidget, IsAnimated, HasTrigger, HasPlacement, HasText, HasShowDelay, HasVisibility {
044
045        Widget widget;
046
047        /**
048         * Whether the widget is animated or not.
049         */
050        protected boolean animated = true;
051
052        /**
053         * The placement of the widget relative to its trigger element.
054         */
055        protected Placement placement = Placement.TOP;
056
057        /**
058         * The action that triggers the widget.
059         */
060        protected Trigger trigger = Trigger.HOVER;
061
062        /**
063         * The delay until the widget is shown.
064         */
065        protected int showDelayInMilliseconds = 0;
066
067        /**
068         * The delay until the widget is hidden.
069         */
070        protected int hideDelayInMilliseconds = 0;
071
072        /**
073         * Creates a new widget based on the provided HTML tag.
074         */
075        public HoverBase() {
076        }
077
078        /**
079         * {@inheritDoc}
080         */
081        @Override
082        public Widget asWidget() {
083                
084            if(getWidget() != null) {
085                removeDataIfExists();
086                
087                reconfigure();
088                
089                    getWidget().addAttachHandler(new AttachEvent.Handler() {
090                        
091                        @Override
092                        public void onAttachOrDetach(AttachEvent event) {
093                            if (!event.isAttached()) {
094                                changeVisibility(VisibilityChange.HIDE);
095                            }
096                        }
097                    });
098                }
099                
100                return getWidget();
101
102        }
103        
104        protected void removeDataIfExists() {
105                removeDataIfExists(getWidget().getElement(), getDataName());
106        }
107        
108        protected native void removeDataIfExists(Element e, String dataName) /*-{
109                if($wnd.jQuery(e).data(dataName)) {
110                        var data = $wnd.jQuery(e).data(dataName);
111                        var eventIn, eventOut;
112                        if (data.options.trigger != 'manual') {
113                                eventIn  = data.options.trigger == 'hover' ? 'mouseenter' : 'focus'
114                                eventOut = data.options.trigger == 'hover' ? 'mouseleave' : 'blur'
115                                data.$element.off(eventIn);
116                                data.$element.off(eventOut);
117                        }
118                        $wnd.jQuery(e).removeData(dataName);
119                }
120        }-*/;
121
122        /**
123         * Adds an HTML data attribute to the widget's tag.
124         * 
125         * @param e target element
126         * 
127         * @param attribute
128         *            the name of the attribute without leading <code>"data-"</code>
129         * @param value
130         *            the value to be stored
131         */
132        protected void setDataAttribute(Element e , String attribute, String value) {
133                e.setAttribute("data-" + attribute, value);
134        }
135
136        /**
137         * {@inheritDoc}
138         */
139        public void setAnimation(boolean animated) {
140                this.animated = animated;
141        }
142
143        /**
144         * Redraws the widget with the currently set options. This must <b>not</b>
145         * be called when a parameter is updated because it would deactivate all
146         * other parameters. No idea why...
147         */
148        public abstract void reconfigure();
149
150        /**
151         * {@inheritDoc}
152         */
153        public boolean getAnimation() {
154                return animated;
155        }
156
157        /**
158         * {@inheritDoc} Relative to its trigger element.
159         */
160        public void setPlacement(Placement placement) {
161                this.placement = placement;
162        }
163
164        /**
165         * {@inheritDoc}
166         */
167        public Placement getPlacement() {
168                return placement;
169        }
170
171        /**
172         * {@inheritDoc}
173         */
174        public void setTrigger(Trigger trigger) {
175                this.trigger = trigger;
176        }
177
178        /**
179         * {@inheritDoc}
180         */
181        public Trigger getTrigger() {
182                return trigger;
183        }
184
185        /**
186         * {@inheritDoc}
187         */
188        public void setShowDelay(int delayInMilliseconds) {
189                showDelayInMilliseconds = delayInMilliseconds;
190        }
191
192        /**
193         * {@inheritDoc}
194         */
195        public int getShowDelay() {
196                return showDelayInMilliseconds;
197        }
198
199        /**
200         * {@inheritDoc}
201         */
202        public void setHideDelay(int delayInMilliseconds) {
203                hideDelayInMilliseconds = delayInMilliseconds;
204        }
205
206        /**
207         * {@inheritDoc}
208         */
209        public int getHideDelay() {
210                return hideDelayInMilliseconds;
211        }
212
213        /**
214         * {@inheritDoc}
215         */
216        public void show() {
217                changeVisibility(VisibilityChange.SHOW);
218        }
219
220        /**
221         * {@inheritDoc}
222         */
223        public void hide() {
224                changeVisibility(VisibilityChange.HIDE);
225        }
226
227        /**
228         * {@inheritDoc}
229         */
230        public void toggle() {
231                changeVisibility(VisibilityChange.TOGGLE);
232        }
233
234        /**
235         * Changes the visibility of the widget.
236         * 
237         * @param visibilityChange
238         *            the action to be performed
239         */
240        protected abstract void changeVisibility(VisibilityChange visibilityChange);
241
242        /**
243         * Adds a widget to this panel.
244         * 
245         * @param w
246         *            the child widget to be added
247         */
248        @Override
249        public void add(Widget w) {
250                // Can't add() more than one widget to a SimplePanel.
251                if (getWidget() != null) {
252                        throw new IllegalStateException("SimplePanel can only contain one child widget");
253                }
254                setWidget(w);
255        }
256
257        /**
258         * Gets the panel's child widget.
259         * 
260         * @return the child widget, or <code>null</code> if none is present
261         */
262        public Widget getWidget() {
263                return widget;
264        }
265
266        public Iterator<Widget> iterator() {
267                // Return a simple iterator that enumerates the 0 or 1 elements in this
268                // panel.
269                return new Iterator<Widget>() {
270
271                        boolean hasElement = widget != null;
272
273                        Widget returned = null;
274
275                        public boolean hasNext() {
276                                return hasElement;
277                        }
278
279                        public Widget next() {
280                                if (!hasElement || (widget == null)) {
281                                        throw new NoSuchElementException();
282                                }
283                                hasElement = false;
284                                return (returned = widget);
285                        }
286
287                        public void remove() {
288                                if (returned != null) {
289                                        HoverBase.this.remove(returned);
290                                }
291                        }
292                };
293        }
294
295        @Override
296        public boolean remove(Widget w) {
297                // Validate.
298                if (widget != w) {
299                        return false;
300                }
301
302                // Logical detach.
303                widget = null;
304                return true;
305        }
306
307        public void setWidget(IsWidget w) {
308                setWidget(w == null ? null : w.asWidget());
309        }
310        
311        /**
312         * Sets this panel's widget. Any existing child widget will be removed.
313         * 
314         * @param w
315         *            the panel's new widget, or <code>null</code> to clear the
316         *            panel
317         */
318        public void setWidget(Widget w) {
319                // Validate
320                if (w == widget) {
321                        return;
322                }
323                
324                widget = w;
325        }
326        
327        @Override
328        public void clear() {
329            widget = null;
330        }
331
332        /**
333         * Get data name of JS Data API.
334         * @return data name
335         */
336        protected abstract String getDataName();
337}
