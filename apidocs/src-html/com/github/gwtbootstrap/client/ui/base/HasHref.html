001    package com.github.gwtbootstrap.client.ui.base;
002    
003    public interface HasHref {
004    
005            /**
006             * Sets the <code>href</code> property of the anchor tag.
007             * <p>
008             * Use it only to set links to another page. Links on the same page or GWT
009             * tokens should be set with {@code #setTargetHistoryToken(String)}
010             * 
011             * @param href
012             *            the website URL
013             */
014            void setHref(String href);
015    
016            /**
017             * @return the URL of the website this link is pointing to
018             */
019            String getHref();
020    
021            /**
022             * Sets the <code>href</code> property of the anchor tag to
023             * <code>"#" + targetHistoryToken</code>.
024             * 
025             * @param targetHistoryToken
026             *            the history token
027             */
028            void setTargetHistoryToken(String targetHistoryToken);
029    
030            /**
031             * Gets the history token this link points to
032             * 
033             * @return the history token
034             */
035            String getTargetHistoryToken();
036    
037    }
