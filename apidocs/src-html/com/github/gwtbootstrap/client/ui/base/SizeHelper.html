001package com.github.gwtbootstrap.client.ui.base;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006
007import com.github.gwtbootstrap.client.ui.constants.Constants;
008import com.google.gwt.user.client.ui.UIObject;
009
010/**
011 * THe helper for {@link HasSize} interface.
012 * 
013 * @author ohashi keisuke
014 * 
015 */
016public class SizeHelper {
017
018        private static List<SizeSpan> SIZE_LIST;
019
020        // create SIZE_LIST
021        static {
022
023                List<SizeSpan> list = new ArrayList<SizeSpan>();
024
025                for (int i = Constants.MINIMUM_SPAN_SIZE; i <= Constants.MAXIMUM_SPAN_SIZE; i++) {
026                        list.add(new SizeSpan(i));
027                }
028
029                SIZE_LIST = Collections.unmodifiableList(list);
030        }
031
032        /**
033         * change size style
034         * 
035         * @param widget
036         *            target widget
037         * @param size
038         *            size
039         */
040        public static void setSize(UIObject widget, int size) {
041                StyleHelper.changeStyle(widget,
042                                                                new SizeSpan(size),
043                                                                SIZE_LIST.toArray(new SizeSpan[SIZE_LIST.size()])
044                                                );
045        }
046
047        private static class SizeSpan implements Style {
048
049                private static final String SIZE_ERROR_MESSAGE = "The size of the Column has to be between " + Constants.MINIMUM_SPAN_SIZE + " and " + Constants.MAXIMUM_SPAN_SIZE + "!";
050
051                private final int size;
052
053                private SizeSpan(int size) {
054
055                        if (size < Constants.MINIMUM_SPAN_SIZE)
056                                throw new IllegalArgumentException(SIZE_ERROR_MESSAGE);
057
058                        if (size > Constants.MAXIMUM_SPAN_SIZE)
059                                throw new IllegalArgumentException(SIZE_ERROR_MESSAGE);
060
061                        this.size = size;
062                }
063
064                @Override
065                public String get() {
066                        return Constants.SPAN + size;
067                }
068
069        }
070
071}
