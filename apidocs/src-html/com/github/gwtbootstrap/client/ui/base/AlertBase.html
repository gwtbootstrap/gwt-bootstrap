016package com.github.gwtbootstrap.client.ui.base;
017
018import com.github.gwtbootstrap.client.ui.Close;
019import com.github.gwtbootstrap.client.ui.event.CloseEvent;
020import com.github.gwtbootstrap.client.ui.event.CloseHandler;
021import com.github.gwtbootstrap.client.ui.event.ClosedEvent;
022import com.github.gwtbootstrap.client.ui.event.ClosedHandler;
023import com.github.gwtbootstrap.client.ui.event.HasCloseHandlers;
024import com.github.gwtbootstrap.client.ui.resources.Bootstrap;
025import com.github.gwtbootstrap.client.ui.resources.Bootstrap.AlertStyle;
026import com.google.gwt.dom.client.Element;
027import com.google.gwt.event.shared.HandlerRegistration;
028import com.google.gwt.safehtml.shared.SafeHtmlUtils;
029import com.google.gwt.user.client.ui.HasHTML;
030
031/**
032 * Base class for Alert widgets.
033 * 
034 * @since 2.0.2.0
035 * 
036 * @author Dominik Mayer
037 * @author Carlos Alexandro Becker
038 * 
039 * @see <a
040 *      href="http://twitter.github.com/bootstrap/components.html#alerts">Bootstrap
041 *      documentation</a>
042 */
043public abstract class AlertBase extends DivWidget implements HasHTML,
044                IsAnimated, HasCloseHandlers, HasType {
045
046        private Close close;
047
048        private String html = "";
049
050        private String heading = "";
051
052        private boolean fade;
053
054        /**
055         * Initializes an Alert with a close icon.
056         */
057        public AlertBase() {
058                this(true);
059        }
060
061        /**
062         * Initializes an Alert with an optional close icon.
063         * 
064         * @param hasClose
065         *            whether the Alert should have a close icon.
066         */
067        public AlertBase(boolean hasClose) {
068                super.setStyle(AlertStyle.ALERT);
069                setClose(hasClose);
070                setHandlerFunctions();
071        }
072
073        /**
074         * Sets whether the Alert has a close icon or not.
075         * 
076         * @param hasClose
077         *            <code>false</code> if you don't want to have a close icon.
078         *            Default: <code>true</code>
079         */
080        public void setClose(boolean hasClose) {
081                if (hasClose) {
082                        close = new Close(Close.DataDismiss.ALERT);
083                        add(close);
084                } else {
085                        clear();
086                        close = null;
087                }
088                redraw();
089        }
090
091        /**
092         * Adds the Java functions that fire the Events to document. It is a
093         * convenience method to have a cleaner code later on.
094         */
095        private native void setHandlerFunctions() /*-{
096                var that = this;
097                $wnd.fireCloseEvent = $entry(function() {
098                        that.@com.github.gwtbootstrap.client.ui.base.AlertBase::fireCloseEvent()();
099                });
100                $wnd.fireClosedEvent = $entry(function() {
101                        that.@com.github.gwtbootstrap.client.ui.base.AlertBase::fireClosedEvent()();
102                });
103        }-*/;
104
105        private void fireCloseEvent() {
106                fireEvent(new CloseEvent());
107        }
108
109        private void fireClosedEvent() {
110                fireEvent(new ClosedEvent());
111        }
112
113        /**
114         * Initializes an Alert with a close icon and the given style.
115         * 
116         * @param style
117         *            of the Alert
118         */
119        public AlertBase(AlertStyle style) {
120                this();
121                setType(style);
122        }
123
124        /**
125         * Sets the type of the Alert.
126         * 
127         * @param style
128         */
129        public void setType(Style style) {
130                super.setStyle(Bootstrap.AlertStyle.ALERT);
131                super.addStyle(style);
132        }
133
134        /**
135         * Sets the type of the Alert based on the Bootstrap class name.
136         * 
137         * @param typeName
138         *            name of the class to add without the leading "alert-".
139         * 
140         * @deprecated This method should never be called directly. It will break
141         *             your implementation if any style names change. The only valid
142         *             use is inside UiBinder files where it processes the
143         *             <code>style="..."</code> argument. Use
144         *             {@link #setType(Style)} instead!
145         */
146        @Deprecated
147        public void setType(String typeName) {
148                if (typeName.equalsIgnoreCase("error"))
149                        setType(Bootstrap.AlertStyle.ERROR);
150                else if (typeName.equalsIgnoreCase("success"))
151                        setType(Bootstrap.AlertStyle.SUCCESS);
152                else if (typeName.equalsIgnoreCase("info"))
153                        setType(Bootstrap.AlertStyle.INFO);
154
155                setFade();
156        }
157
158        /**
159         * Sets the text of an optional heading. The implementation depends on the
160         * subclass.
161         * 
162         * @param text
163         *            the new heading
164         */
165        public void setHeading(String text) {
166                heading = text;
167                redraw();
168        }
169
170        /**
171         * Sets whether the Alert should be animated.
172         * 
173         * @param animated
174         *            <code>true</code> if the Alert should fade out. Default:
175         *            <code>false</code>
176         */
177        public void setAnimated(boolean animated) {
178                this.fade = animated;
179                setFade();
180        }
181
182        /**
183         * Delete the whole content of the Alert. This includes text, heading and
184         * close icon.
185         */
186        @Override
187        public void clear() {
188                getElement().setInnerHTML("");
189        }
190
191        /**
192         * Redraws the Alert according to the specified settings.
193         */
194        protected void redraw() {
195                setHTML(html);
196                configure(getElement());
197        }
198
199        /**
200         * Sets the classes that define whether the Alert fades or not.
201         */
202        private void setFade() {
203                if (fade) {
204                        addStyleName("fade");
205                        addStyleName("in");
206                } else {
207                        removeStyleName("fade");
208                        removeStyleName("in");
209                }
210        }
211
212        /**
213         * {@inheritDoc}
214         */
215        public String getText() {
216                return getHTML(); // TODO
217        }
218
219        /**
220         * {@inheritDoc}
221         */
222        public void setText(String text) {
223                setHTML(SafeHtmlUtils.htmlEscape(text));
224        }
225
226        /**
227         * {@inheritDoc}
228         */
229        public String getHTML() {
230                return html;
231        }
232
233        /**
234         * {@inheritDoc}
235         */
236        public void setHTML(String html) {
237                String output = "";
238                if (close != null)
239                        output = close.toString();
240                output = output + heading + html;
241                setHTMLOutput(html, output);
242        }
243
244        /**
245         * Sets the HTML of the Alert. Override this method to implement a custom
246         * formatting.
247         * 
248         * @param html
249         *            the html of the content without heading and close icon.
250         * @param output
251         *            the raw output to be put into the widget. Be careful!
252         */
253        protected void setHTMLOutput(String html, String output) {
254                this.html = html;
255                getElement().setInnerHTML(output);
256        }
257
258        /**
259         * {@inheritDoc}
260         */
261        @Override
262        protected void onLoad() {
263                super.onLoad();
264                configure(getElement());
265        }
266
267        private native void configure(Element e) /*-{
268                $wnd.jQuery(e).alert(e);
269        }-*/;
270
271        /**
272         * {@inheritDoc}
273         */
274        public HandlerRegistration addCloseHandler(CloseHandler handler) {
275                addCloseHandler(getElement());
276                return addHandler(handler, CloseEvent.getType());
277        }
278
279        private native void addCloseHandler(Element e) /*-{
280                $wnd.jQuery(e).bind('close', function() {
281                        $wnd.fireCloseEvent();
282                });
283        }-*/;
284
285        /**
286         * {@inheritDoc}
287         */
288        public HandlerRegistration addClosedHandler(ClosedHandler handler) {
289                addClosedHandler(getElement());
290                return addHandler(handler, ClosedEvent.getType());
291        }
292
293        private native void addClosedHandler(Element e) /*-{
294                $wnd.jQuery(e).bind('closed', function() {
295                        $wnd.fireClosedEvent();
296                });
297        }-*/;
298}
