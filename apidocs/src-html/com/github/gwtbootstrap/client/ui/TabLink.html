001    package com.github.gwtbootstrap.client.ui;
002    
003    import com.github.gwtbootstrap.client.ui.base.IconAnchor;
004    import com.github.gwtbootstrap.client.ui.constants.Constants;
005    import com.google.gwt.user.client.Element;
006    
007    public class TabLink extends NavLink {
008    
009        private TabPane pane;
010    
011        public TabLink(TabPane pane) {
012            this();
013            setText(pane.getHeading());
014            setTablePane(pane);
015        }
016    
017        public void setTablePane(TabPane pane) {
018            this.pane = pane;
019    
020            IconAnchor anchor = getAnchor();
021            anchor.getElement().setAttribute(Constants.DATA_TOGGLE, "tab");
022            anchor.getElement().setAttribute(Constants.DATA_TARGET,
023                    "#" + pane.getId());
024            if (pane.isActive() || this.isActive()) {
025                show();
026            }
027        }
028        
029        public TabPane getTabPane() {
030            return pane;
031        }
032        
033        public TabLink() {
034            super();
035        }
036    
037        public void show() {
038            setActive(true);
039            show(getAnchor().getElement());
040        }
041    
042        //@formatter:off
043        public native void show(Element e)/*-{
044            $wnd.jQuery(e).tab('show');
045        }-*/;
046        //@formatter:on
047    
048    }
