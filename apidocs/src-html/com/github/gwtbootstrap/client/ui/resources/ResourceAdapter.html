016package com.github.gwtbootstrap.client.ui.resources;
017
018import com.github.gwtbootstrap.client.ui.base.IsResponsive;
019
020/**
021 * <h3>Using custom css/js resources.</h3>
022 * 
023 * <p>
024 * If you need to adapt resources, implement this interface, and add a
025 * <code>replace-with</code> tag to your module xml. Example:
026 * </p>
027 * 
028 * <p>
029 * 1. Create a Resources Interface (extending {@link Resources}) override the
030 * getters of the files you want to replace.
031 * 
032 * <pre>
033 * {@code
034 *      public interface MyResources extends Resources {
035 *              {@literal @}Source("mycss/custom-bootstrap.min.css")
036 *              TextResource bootstrapCss();
037 *      }}
038 * </pre>
039 * 
040 * </p>
041 * 
042 * <p>
043 * 2. Create a <code>ResourceAdapter</code>.
044 * 
045 * <pre>
046 * {@code
047 *      public MyResourceAdapter implements ResourceAdapter {
048 *              public Resources getResources() {
049 *                      return GWT.create(MyResources.class);
050 *              }
051 *      }}
052 * </pre>
053 * 
054 * </p>
055 * 
056 * <p>
057 * 3. Add a <code>replace-with</code> tag to your module xml (
058 * <code>*.gwt.xml</code>).
059 * 
060 * <pre>
061 * {@code
062 * <replace-with class="userpackage.MyResourceAdapter">
063 *       <when-type-is class="com.github.gwtbootstrap.client.ui.resources.ResourceAdapter" />
064 * </replace-with>
065 * }
066 * </pre>
067 * 
068 * </p>
069 * 
070 * @since 2.0.2.0
071 * 
072 * @author soundTricker
073 * @author Dominik Mayer
074 * 
075 * @see Resources
076 * @see ResourceAdapterImpl
077 */
078public interface ResourceAdapter {
079
080        /**
081         * Get the Bootstrap Resources that should be used for this project.
082         * 
083         * @return the Bootstrap Resources
084         */
085        Resources getResources();
086
087        //@formatter:off
088        /**
089         * Determines whether the project uses a responsive design.
090         * 
091         * <p>
092         * If the responsive
093         * design is enabled, the interface can adapt to the screen size and show
094         * different content on smartphones, tablets and desktop computers.
095         * </p>
096         * 
097         * @return <code>true</code> if the responsive features should be enabled.
098         * Default: <code>false</code>
099         * 
100         * @see IsResponsive
101         * 
102         * @see <a href="http://twitter.github.com/bootstrap/scaffolding.html#responsive">Bootstrap documentation</a>
103         */
104        boolean hasResponsiveDesign();
105}
