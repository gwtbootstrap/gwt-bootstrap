016    package com.github.gwtbootstrap.client.ui.resources;
017    
018    import com.google.gwt.dom.client.Document;
019    import com.google.gwt.dom.client.HeadElement;
020    import com.google.gwt.dom.client.ScriptElement;
021    
022    /**
023     * 
024     * @author Carlos Alexandro Becker
025     * @since 22/01/2012
026     */
027    public class JavaScriptInjector extends AbstractInjector {
028    
029            public static void inject(String javascript) {
030                    HeadElement head = getHead();
031                    ScriptElement element = createScriptElement();
032                    element.setText(javascript);
033                    head.appendChild(element);
034            }
035    
036            private static ScriptElement createScriptElement() {
037                    ScriptElement script = Document.get().createScriptElement();
038                    script.setAttribute("type", "text/javascript");
039                    return script;
040            }
041    
042    }
