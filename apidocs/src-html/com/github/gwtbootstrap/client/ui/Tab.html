001package com.github.gwtbootstrap.client.ui;
002
003import java.util.Iterator;
004
005import com.github.gwtbootstrap.client.ui.base.HasIcon;
006import com.github.gwtbootstrap.client.ui.base.HasStyle;
007import com.github.gwtbootstrap.client.ui.base.IsResponsive;
008import com.github.gwtbootstrap.client.ui.base.Style;
009import com.github.gwtbootstrap.client.ui.constants.Device;
010import com.github.gwtbootstrap.client.ui.constants.IconSize;
011import com.github.gwtbootstrap.client.ui.constants.IconType;
012import com.google.gwt.event.dom.client.ClickHandler;
013import com.google.gwt.event.dom.client.HasClickHandlers;
014import com.google.gwt.event.shared.GwtEvent;
015import com.google.gwt.event.shared.HandlerRegistration;
016import com.google.gwt.user.client.DOM;
017import com.google.gwt.user.client.ui.HasWidgets;
018import com.google.gwt.user.client.ui.IsWidget;
019import com.google.gwt.user.client.ui.Widget;
020
021/**
022 * The tab widget for {@link TabPanel}.
023 * 
024 * It's for only UiBinder.
025 * 
026 * @author ohashi keisuke
027 *
028 */
029public class Tab implements IsWidget, HasWidgets, HasClickHandlers,HasStyle, IsResponsive,HasIcon {
030    
031    TabLink link = new TabLink();
032
033    public Tab() {
034        TabPane tabPane = new TabPane();
035        
036        tabPane.setHref(DOM.createUniqueId());
037        
038        link.setTablePane(tabPane);
039    }
040    
041    @Override
042    public Widget asWidget() {
043        return link;
044    }
045    
046    protected TabPane getTabPane() {
047        return link.getTabPane();
048    }
049    
050    public TabLink asTabLink() {
051        return link;
052    }
053    
054    public void setActive(boolean active) {
055        link.setActive(active);
056    }
057    
058    public void setHeading(String text) {
059        link.setText(text);
060    }
061    
062    public String getHeading() {
063        return link.getText();
064    }
065    
066    @Override
067    public void add(Widget w) {
068        link.getTabPane().add(w);
069    }
070
071    @Override
072    public void clear() {
073        link.getTabPane().clear();
074    }
075
076    @Override
077    public Iterator<Widget> iterator() {
078        return link.getTabPane().iterator();
079    }
080
081    @Override
082    public boolean remove(Widget w) {
083        return link.getTabPane().remove(w);
084    }
085
086    @Override
087    public HandlerRegistration addClickHandler(ClickHandler handler) {
088        return link.addClickHandler(handler);
089    }
090
091    @Override
092    public void setIcon(IconType type) {
093        link.setIcon(type);
094    }
095
096    @Override
097    public void setIconSize(IconSize size) {
098        link.setIconSize(size);
099    }
100
101    @Override
102    public void setShowOn(Device device) {
103        link.setShowOn(device);
104        link.getTabPane().setShowOn(device);
105    }
106
107    @Override
108    public void setHideOn(Device device) {
109        link.setHideOn(device);
110        link.getTabPane().setHideOn(device);
111    }
112
113    @Override
114    public void setStyle(Style style) {
115        link.setStyle(style);
116    }
117
118    @Override
119    public void addStyle(Style style) {
120        link.addStyle(style);
121    }
122
123    @Override
124    public void removeStyle(Style style) {
125        link.removeStyle(style);
126    }
127
128    @Override
129    public void fireEvent(GwtEvent<?> event) {
130        link.fireEvent(event);
131    }
132}
