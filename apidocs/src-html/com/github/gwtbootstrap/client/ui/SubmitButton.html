001    package com.github.gwtbootstrap.client.ui;
002    
003    import com.github.gwtbootstrap.client.ui.base.HasType;
004    import com.github.gwtbootstrap.client.ui.base.StyleHelper;
005    import com.github.gwtbootstrap.client.ui.constants.ButtonType;
006    import com.github.gwtbootstrap.client.ui.constants.Constants;
007    import com.github.gwtbootstrap.client.ui.resources.ButtonSize;
008    import com.google.gwt.dom.client.Document;
009    import com.google.gwt.dom.client.Element;
010    import com.google.gwt.dom.client.InputElement;
011    import com.google.gwt.event.dom.client.ChangeEvent;
012    import com.google.gwt.event.dom.client.ChangeHandler;
013    import com.google.gwt.event.dom.client.ClickHandler;
014    import com.google.gwt.event.dom.client.HasChangeHandlers;
015    import com.google.gwt.event.logical.shared.ValueChangeEvent;
016    import com.google.gwt.event.logical.shared.ValueChangeHandler;
017    import com.google.gwt.event.shared.HandlerRegistration;
018    import com.google.gwt.safehtml.shared.SafeHtml;
019    import com.google.gwt.user.client.ui.FocusWidget;
020    import com.google.gwt.user.client.ui.HasName;
021    import com.google.gwt.user.client.ui.HasText;
022    import com.google.gwt.user.client.ui.HasValue;
023    
024    /**
025     * A SubmitButton for Bootstrap form.
026     * 
027     * @since 2.0.3.0
028     * @author ohashi keisuke
029     */
030    public class SubmitButton extends FocusWidget implements HasType<ButtonType>, HasValue<String>, HasText, HasName, HasChangeHandlers {
031    
032            private boolean valueChangeHandlerInitialized;
033    
034            /**
035             * Creates an submit button with no caption.
036             */
037            public SubmitButton() {
038                    super(Document.get().createSubmitInputElement());
039                    setStyleName(Constants.BTN);
040            }
041    
042            private InputElement asInputElement() {
043                    return getElement().cast();
044            }
045    
046            /**
047             * This constructor may be used by subclasses to explicitly use an existing
048             * element. This element must be a &lt;input type="submit"&gt; element.
049             * 
050             * @param element
051             *            the element to be used
052             */
053            protected SubmitButton(Element element) {
054                    super(element);
055                    assert "submit".equalsIgnoreCase(element.<InputElement> cast().getType());
056            }
057    
058            /**
059             * Creates a button with the given HTML caption and click listener.
060             * 
061             * @param html
062             *            the HTML caption
063             * @param handler
064             *            the click handler
065             */
066            public SubmitButton(SafeHtml html,
067                    ClickHandler handler) {
068                    this(html.asString(), handler);
069            }
070    
071            /**
072             * Creates a button with the given HTML caption.
073             * 
074             * @param html
075             *            the HTML caption
076             */
077            public SubmitButton(SafeHtml html) {
078                    this(html.asString());
079            }
080    
081            /**
082             * Creates a button with the given Text caption and click listener.
083             * 
084             * @param text
085             *            the Text caption
086             * @param handler
087             *            the click handler
088             */
089            public SubmitButton(String text,
090                    ClickHandler handler) {
091                    this(text);
092                    addClickHandler(handler);
093            }
094    
095            /**
096             * Creates a button with the given Text caption.
097             * 
098             * @param text
099             *            the Text caption
100             */
101            public SubmitButton(String text) {
102                    this();
103                    setValue(text);
104            }
105    
106            /**
107             * Programmatic equivalent of the user clicking the button.
108             */
109            public void click() {
110                    getElement().<InputElement> cast().click();
111            }
112    
113            /**
114             * Sets the type of the Button.
115             * <p>
116             * Different types give the button a different look.
117             * 
118             * @param type
119             *            the type of the Button.
120             */
121            public void setType(ButtonType type) {
122                    StyleHelper.changeStyle(this, type, ButtonType.class);
123            }
124    
125            /**
126             * Sets the size of the Button.
127             * 
128             * @param size
129             *            the size of the Button.
130             */
131            public void setSize(ButtonSize size) {
132                    StyleHelper.changeStyle(this, size, ButtonSize.class);
133            }
134    
135            /**
136             * {@inheritDoc}
137             */
138            public HandlerRegistration addChangeHandler(ChangeHandler handler) {
139                    return addDomHandler(handler, ChangeEvent.getType());
140            }
141    
142            /**
143             * @see com.google.gwt.event.logical.shared.HasValueChangeHandlers#addValueChangeHandler(com.google.gwt.event.logical.shared.ValueChangeHandler)
144             */
145            public HandlerRegistration addValueChangeHandler(ValueChangeHandler<String> handler) {
146                    if (!valueChangeHandlerInitialized) {
147                            valueChangeHandlerInitialized = true;
148                            addChangeHandler(new ChangeHandler() {
149    
150                                    public void onChange(ChangeEvent event) {
151                                            ValueChangeEvent.fire(SubmitButton.this, getValue());
152                                    }
153                            });
154                    }
155                    return addHandler(handler, ValueChangeEvent.getType());
156            }
157    
158            /**
159             * {@inheritDoc}
160             */
161            @Override
162            public void setName(String name) {
163                    asInputElement().setName(name);
164            }
165    
166            /**
167             * {@inheritDoc}
168             */
169            @Override
170            public String getName() {
171                    return asInputElement().getName();
172            }
173    
174            /**
175             * {@inheritDoc}
176             */
177            @Override
178            public String getText() {
179                    return asInputElement().getValue();
180            }
181    
182            /**
183             * {@inheritDoc}
184             */
185            @Override
186            public void setText(String text) {
187                    setValue(text);
188            }
189    
190            /**
191             * {@inheritDoc}
192             */
193            @Override
194            public String getValue() {
195                    return asInputElement().getValue();
196            }
197    
198            /**
199             * {@inheritDoc}
200             */
201            @Override
202            public void setValue(String value) {
203                    this.setValue(value , false);
204            }
205    
206            /**
207             * {@inheritDoc}
208             */
209            @Override
210            public void setValue(String value, boolean fireEvents) {
211                    String oldValue = getValue();
212                    asInputElement().setValue(value);
213                    if (fireEvents) {
214                            ValueChangeEvent.fireIfNotEqual(this, oldValue, value);
215                    }
216            }
217    }
