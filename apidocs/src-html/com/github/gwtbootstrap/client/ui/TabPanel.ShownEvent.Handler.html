209     * Create an empty TabPanel.

212    public TabPanel(Bootstrap.Tabs position) {
213        setTabPosition(position.get().toLowerCase());
214        setHandlerFunctions(getElement());
215    }

217    /**
218     * Set tab position
219     * @param position tab position.
220     */
221    public void setTabPosition(String position) {
222        if (tabs.getParent() != null) {
223            remove(tabs);
224            remove(tabContent);
225        }
226
227        if (position.equalsIgnoreCase("below")) {
228            setStyle(Bootstrap.Tabs.BELOW);
229            super.add(tabContent);
230            super.add(tabs);
231        } else if (position.equalsIgnoreCase("left")) {
232            setStyle(Bootstrap.Tabs.LEFT);
233            super.add(tabs);
234            super.add(tabContent);
235        } else if (position.equalsIgnoreCase("right")) {
236            setStyle(Bootstrap.Tabs.RIGHT);
237            super.add(tabs);
238            super.add(tabContent);
239        } else {
240            setStyle(Bootstrap.Tabs.ABOVE);
241            super.add(tabs);
242            super.add(tabContent);
243        }
244    }

246    @Override
247    public void add(Widget child) {
248
249        if (child instanceof TabPane) {
250            add((TabPane) child);
251            return;
252        }
253
254        if (child instanceof TabLink) {
255            add((TabLink) child);
256            return;
257        }
258
259        if (child instanceof DropdownTab) {
260            add((DropdownTab) child);
261            return;
262        }
263
264        if (GWT.isProdMode()) {
265            throw new IllegalArgumentException(
266                    "TabPanel can add only TabPane or TabLink or Tab or DorpdownTab. you added "
267                            + child);
268        }
269    }
270
271    private void add(DropdownTab dropdownTab) {
272
273        tabs.add(dropdownTab);
274
275        List<Tab> tabList = dropdownTab.getTabList();
276        for (Tab tab : tabList) {
277            tabLinkList.add(tab.asTabLink());
278            TabPane tabPane = tab.getTabPane();
279            tabContent.add(tabPane);
280        }
281    }
282
283    private void add(TabPane child) {
284
285        if (child.isCreateTabLink()) {
286            TabLink tabLink = new TabLink(child);
287            tabs.add(tabLink);
288            tabLinkList.add(tabLink);
289        }
290        tabContent.add(child);
291    }
292
293    private void add(final TabLink child) {
294
295        if (child.isCreateTabPane() && child.getTabPane() == null) {
296            TabPane pane = new TabPane(child.getText());
297            child.setTablePane(pane);
298            tabContent.add(pane);
299        } else if (child.getTabPane() != null) {
300            tabContent.add(child.getTabPane());
301        }
302        tabs.add(child);
303        tabLinkList.add(child);
304    }
305
306    @Override
307    public void clear() {
308        tabContent.clear();
309        tabs.clear();
310        tabLinkList.clear();
311    }

313    /**
314     * Remove tab or tabpane.
315     * <p>
316     * If Tablink has TabPane,romve TabPane with TabLink. </pre> {@inheritDoc}
317     */
318    @Override
319    public boolean remove(int index) {
320        Widget widget = tabs.getWidget(index);
321
322        if (widget instanceof TabLink) {
323            TabLink link = (TabLink) widget;
324            if (link.getTabPane() != null) {
325                link.getTabPane().removeFromParent();
326            }
327            tabLinkList.remove(link);
328            return tabs.remove(index);
329        } else if (widget instanceof DropdownTab) {
330
331            DropdownTab dropdownTab = (DropdownTab) widget;
332
333            List<Tab> tabList = dropdownTab.getTabList();
334
335            for (Tab tab : tabList) {
336                tabLinkList.remove(tab.asTabLink());
337                if (tab.getTabPane() != null) {
338                    tab.getTabPane().removeFromParent();
339                }
340            }
341            return tabs.remove(dropdownTab);
342        } else if (widget instanceof TabPane) {
343
344            return tabContent.remove(widget);
345        }
346
347        return super.remove(widget);
348    }
349
350    /**
351     * remove TabLink or TabPane.
352     * <p>
353     * </p>
354     * {@inheritDoc}
355     */
356    @Override
357    public boolean remove(Widget w) {
358
359        if (w instanceof TabLink) {
360            TabLink link = (TabLink) w;
361            tabLinkList.remove(link);
362            if (link.getTabPane() != null) {
363                link.getTabPane().removeFromParent();
364            }
365            return tabs.remove(w);
366        } else if (w instanceof DropdownTab) {
367            DropdownTab dropdownTab = (DropdownTab) w;
368
369            List<Tab> tabList = dropdownTab.getTabList();
370
371            for (Tab tab : tabList) {
372
373                tabLinkList.remove(tab.asTabLink());
374                if (tab.getTabPane() != null) {
375                    tab.getTabPane().removeFromParent();
376                }
377            }
378
379            return tabs.remove(dropdownTab);
380
381        } else if (w instanceof TabPane) {
382            return tabContent.remove(w);
383        }
384
385        return super.remove(w);
386    }
387
388    /**
389     * {@inheritDoc}
390     */
391    @Override
392    public boolean remove(IsWidget child) {
393
394        if (child instanceof Tab) {
395            Tab tab = (Tab) child;
396
397            TabLink link = tab.asTabLink();
398
399            if (link.getTabPane() != null) {
400                link.getTabPane().removeFromParent();
401            }
402            tabLinkList.remove(link);
403            return tabs.remove(link);
404        } else if (child instanceof DropdownTab) {
405            DropdownTab tab = (DropdownTab) child;
406
407            List<Tab> tabList = tab.getTabList();
408
409            for (Tab tab2 : tabList) {
410                tabLinkList.remove(tab2.asTabLink());
411                if (tab2.getTabPane() != null) {
412                    tabContent.remove(tab2.getTabPane());
413                }
414            }
415            return super.remove(child);
416        }
417
418        return super.remove(child);
419    }

429    /**
430     * Get Current selected tab index.
431     * <p>
432     * if not found, return <code>-1.</code>
433     * </p>
434     * 
435     * @return tab index.
436     */
437    public int getSelectedTab() {
438
439        for (int i = 0; i < tabLinkList.size(); i++) {
440            if (tabLinkList.get(i).isActive()) {
441                return i;
442            }
443        }
444        return -1;
445    }

447    private void setHandlerFunctions(final TabLink e) {
448        if (isOrWasAttached()) {
449            setHandlerFunctions(e.getAnchor().getElement());
450            return;
451        }
452
453        Scheduler.get().scheduleDeferred(new ScheduledCommand() {
454
455            @Override
456            public void execute() {
457                setHandlerFunctions(e.getAnchor().getElement());
458            }
459        });
460
461    }

463    protected void onShow(Event e, Element target, Element relatedTarget) {
464        ShowEvent event = new ShowEvent(e);
465        event.setTarget(findTabLink(target));
466        event.setRelatedTarget(findTabLink(relatedTarget));
467        fireEvent(event);
468    }
469
470    protected void onShown(Event e, Element target, Element relatedTarget) {
471        ShownEvent event = new ShownEvent(e);
472        event.setTarget(findTabLink(target));
473        event.setRelatedTarget(findTabLink(relatedTarget));
474        fireEvent(event);
475    }
476
477    public HandlerRegistration addShowHandler(ShowEvent.Handler handler) {
478        return addHandler(handler, ShowEvent.TYPE);
479    }
480
481    public HandlerRegistration addShownHandler(ShownEvent.Handler handler) {
482        return addHandler(handler, ShownEvent.TYPE);
483    }
484
485    private TabLink findTabLink(Element e) {
486        for (TabLink tabLink : tabLinkList)
487            if (tabLink.getAnchor().getElement().equals(e))
488                return tabLink;
489
490        return null;
491    }
