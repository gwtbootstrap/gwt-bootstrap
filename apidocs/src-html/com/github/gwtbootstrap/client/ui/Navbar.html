016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.base.DivWidget;
019import com.github.gwtbootstrap.client.ui.constants.NavbarConstants;
020import com.github.gwtbootstrap.client.ui.constants.NavbarPosition;
021import com.google.gwt.dom.client.BodyElement;
022import com.google.gwt.dom.client.Document;
023import com.google.gwt.dom.client.Style.Unit;
024import com.google.gwt.user.client.Element;
025import com.google.gwt.user.client.ui.RootPanel;
026import com.google.gwt.user.client.ui.Widget;
027
028//@formatter:off
029/**
030 * Navbar with optional text, links and forms.
031 * <p>
032 * The Navbar can be fixed on the top or bottom of the site where it will stay
033 * when the user scrolls the page. For proper rendering, a fixed Navbar cannot
034 * be part of any container! It has to be attached directly to the 
035 * {@link RootPanel}. 
036 * 
037 * <p>
038 * <h3>UiBinder Usage:</h3>
039 * 
040 * <pre>
041 * {@code
042 * <g:FlowPanel>
043 *     <b:Navbar>
044 *         <b:Brand>Bootstrap</b:Brand>
045 *         <b:Nav>
046 *             <b:NavLink>Link 1</b:NavLink>
047 *             <b:NavLink>Link 2</b:NavLink>
048 *         </b:Nav>
049 *         <b:NavForm size="1" />
050 *         <b:Nav alignment="RIGHT">
051 *             <b:NavLink>Link 3</b:NavLink>
052 *         </b:Nav>
053 *     </b:Navbar>
054 *     <b:Container>
055 *         [...]
056 *     </b:Container>
057 * </g:FlowPanel>
058 * 
059 * }
060 * </pre>
061 * </p>
062 * 
063 * @since 2.0.3.0
064 * 
065 * @author Dominik Mayer
066 * @author Carlos Alexandro Becker
067 * 
068 * @see <a href="http://twitter.github.com/bootstrap/components.html#navbar">Bootstrap documentation</a>
069 * @see ResponsiveNavbar
070 */
071//@formatter:on
072public class Navbar extends DivWidget {
073
074        protected static final int TOP_SPACE_IN_PIXEL = 50;
075
076        private boolean fixedTop = false;
077
078        private class NavbarInner extends DivWidget {
079
080                public NavbarInner() {
081                        setStyleName(NavbarConstants.NAVBAR_INNER);
082                }
083        }
084
085        private final Container container = getContainer();
086        private final NavbarInner navbarInner = new NavbarInner();
087        private Scrollspy spy;
088        
089        private boolean scrollspy;
090
091        /**
092         * Creates an empty Navbar.
093         */
094        public Navbar() {
095                setStyleName(NavbarConstants.NAVBAR);
096                navbarInner.add(container);
097                super.add(navbarInner);
098        }
099
100        /**
101         * Defines the default container implementation. You can override this in a
102         * extended class if you want another implementation, like a
103         * {@link FluidContainer}.
104         */
105        protected Container getContainer() {
106                return new Container();
107        }
108
109        /**
110         * Defines whether the Navbar should contain a {@link Scrollspy}.
111         * 
112         * @param scrollspy
113         *            <code>true</code> to include a Scrollspy. Default:
114         *            <code>false</code>
115         */
116        public void setScrollspy(boolean scrollspy) {
117                this.scrollspy = scrollspy;
118        }
119        
120        /**
121         * Defines scrollspy target element.
122         * @param spyElement target element
123         */
124        public void setSpyElement(Element spyElement) {
125                
126                assert spyElement != null;
127                
128                if (spy == null)
129                        spy = new Scrollspy(this);
130                
131                spy.setSpyElement(spyElement);
132                this.scrollspy = true;
133        }
134        
135        @Override
136        protected void onAttach() {
137                super.onAttach();
138                if (spy == null)
139                        spy = new Scrollspy(this);
140
141                if (scrollspy) {
142                        spy.configure();
143                }
144                // TODO make a unconfigure feature.
145        }
146        
147        /**
148         * Fix the Navbar at the top or bottom of the screen.
149         * <p>
150         * For this to work properly, the Navbar must not be a child of a
151         * {@link Container}.
152         * 
153         * @param position
154         *            the position of the Navbar
155         */
156        public void setPosition(NavbarPosition position) {
157                if (position == NavbarPosition.TOP)
158                        fixedTop = true;
159                else
160                        fixedTop = false;
161
162                for (NavbarPosition p : NavbarPosition.values())
163                        if (!p.get().isEmpty())
164                                removeStyleName(p.get());
165
166                addStyle(position);
167                setPaddingTop(true);
168        }
169
170        /**
171         * {@inheritDoc}
172         */
173        @Override
174        public void add(Widget child) {
175                container.add(child);
176        }
177
178        /**
179         * Adds a widget to the Navbar element, <b>not</b> the container.
180         * 
181         * @param widget
182         *            widget to add
183         */
184        protected void addWidget(Widget widget) {
185                super.add(widget);
186        }
187
188        /**
189         * Activates or deactivates a top padding.
190         * <p>
191         * This method can be overridden. The default implementation sets a padding
192         * only when the Navbar is fixed to the top.
193         * 
194         * @param padding
195         */
196        protected void setPaddingTop(boolean padding) {
197                if (padding && fixedTop)
198                        setPaddingTop(TOP_SPACE_IN_PIXEL);
199                else
200                        setPaddingTop(0);
201        }
202
203        private void setPaddingTop(int pixels) {
204                BodyElement body = Document.get().getBody();
205                body.getStyle().setPaddingTop(pixels, Unit.PX);
206        }
207        
208}
