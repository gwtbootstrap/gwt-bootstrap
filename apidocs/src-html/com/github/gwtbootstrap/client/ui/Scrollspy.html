016package com.github.gwtbootstrap.client.ui;
017
018import com.google.gwt.dom.client.Document;
019import com.google.gwt.dom.client.Element;
020
021//@formatter:off
022/**
023 * {@linkplain Navbar} uses this class for the scrollspy.
024 * 
025 * @author Carlos Alexandro Becker
026         * @see <a href="http://twitter.github.com/bootstrap/javascript.html#scrollspy">Twitter Bootstrap ScrollSpy</a>
027 */
028//@formatter:on
029public class Scrollspy {
030
031        private Navbar navbar;
032        
033        /** spy target element */
034        private Element spyElement;
035
036        /** data-target */
037        private String target;
038
039        /** data-offset */
040        private Integer offset;
041        
042        private boolean configured;
043
044        /**
045         * 
046         * @param navbar
047         *            The navbar that will be spied.
048         */
049        public Scrollspy(Navbar navbar) {
050                super();
051                this.navbar = navbar;
052        }
053
054        /**
055         * Get navbar
056         * @return nabvar.
057         */
058        public Navbar getNavbar() {
059                return navbar;
060        }
061
062        /**
063         * set navbar
064         * @param navbar
065         */
066        public void setNavbar(Navbar navbar) {
067                this.navbar = navbar;
068        }
069        
070        /**
071         * set data-target.
072         * <p>
073         * If set this,{@link #spyElement} is set {@code data-target} attribute this {@code target} when {@link #configure()} is called.
074         * </p>
075         * @param target
076         *                               data-target value.like this,{@code #some-id} {@code .some-classes}
077         * 
078         * @see <a href="http://twitter.github.com/bootstrap/javascript.html#scrollspy">Twitter Bootstrap ScrollSpy Options</a>
079         */
080        public void setTarget(String target) {
081                this.target = target;
082                
083        }
084        
085        /**
086         * set spy target element.
087         * <p>
088         * If set this,Scrollspy set some setting to this argment 
089         * when {@link #configure()} is called.<br/>
090         * 
091         * If don't set this,Scrollspy set some setting to {@code body} tag.
092         * </p>
093         * 
094         * @param element
095         *                               spy target element.
096         * @see <a href="http://twitter.github.com/bootstrap/javascript.html#scrollspy">Twitter Bootstrap ScrollSpy Options</a>
097         */
098        public void setSpyElement(Element element) {
099                this.spyElement = element;
100        }
101        
102        /**
103         * set data-offset.
104         * <p>
105         * If set this,{@link #spyElement} is set {@code data-offset} attribute this {@code target} when {@link #configure()} is called.
106         * </p>
107         * @param target
108         *                               data-offset value.
109         * @see <a href="http://twitter.github.com/bootstrap/javascript.html#scrollspy">Twitter Bootstrap ScrollSpy Options</a>
110         */
111        public void setOffset(Integer offset) {
112                this.offset = offset;
113        }
114
115        /**
116         * Configure the scrollspy in the configured navbar.
117         */
118        public void configure() {
119                
120                Element spyTargetElement = getSpyElement();
121                
122                assert spyTargetElement != null : "houston, we need a spy element here!";
123                
124                spyTargetElement.setAttribute("data-spy", "scroll");
125                
126                if(offset != null) {
127                        spyTargetElement.setAttribute("data-offset", String.valueOf(offset));
128                }
129                
130                if(target != null && target.length() > 0) {
131                        spyTargetElement.setAttribute("data-target", target);
132                }
133                
134                jsConfigure(navbar.getElement());
135                
136                configured = true;
137        }
138        
139        public boolean isConfigured() {
140                return configured;
141        }
142        
143        public Element getSpyElement() {
144                return spyElement != null ? spyElement : Document.get().getBody();
145        }
146        
147        private native void jsConfigure(Element e) /*-{
148                $wnd.jQuery(e).scrollspy();
149        }-*/;
150
151}
