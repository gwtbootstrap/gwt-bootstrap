016package com.github.gwtbootstrap.client.ui.constants;
017
018import com.github.gwtbootstrap.client.ui.Form;
019import com.github.gwtbootstrap.client.ui.base.Style;
020
021//@formatter:off
022/**
023* Type of a {@link Form}.
024* 
025* @since 2.0.3.0
026* 
027* @author Dominik Mayer
028* 
029* @see <a href="http://twitter.github.com/bootstrap/base-css.html#forms">Bootstrap documentation</a>
030* 
031*/
032//@formatter:on
033public enum FormType implements Style {
034
035        VERTICAL("vertical"),
036
037        INLINE("inline"),
038
039        SEARCH("search"),
040
041        HORIZONTAL("horizontal");
042
043        private static final String PREFIX = "form-";
044
045        private String className;
046
047        private FormType(String className) {
048                this.className = PREFIX + className;
049        }
050
051        /**
052         * {@inheritDoc}
053         */
054        public String get() {
055                return className;
056        }
057
058}
