001package com.github.gwtbootstrap.client.ui.constants;
002
003import com.github.gwtbootstrap.client.ui.base.Style;
004
005
006/**
007 * input alternate size
008 * @author ohashi keisuke
009 */
010public enum AlternateSize implements Style {
011        MINI("input-mini"),
012        SMALL("input-small"),
013        MEDIUM("input-medium"),
014        LARGE("input-large"),
015        XLARGE("input-xlarge"),
016        XXLARGE("input-xxlarge");
017
018        private AlternateSize(String styleName) {
019                this.styleName = styleName;
020        }
021        
022        private final String styleName;
023        
024        /**
025         * {@inheritDoc}
026         */
027        @Override
028        public String get() {
029                return styleName;
030        }
031
032}
