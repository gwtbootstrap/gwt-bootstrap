006package com.github.gwtbootstrap.client.ui.constants;
007
008import com.github.gwtbootstrap.client.ui.base.Style;
009
010/**
011 * Resize type of {@link com.github.gwtbootstrap.client.ui.TextArea}
012 *
013 * @author Alexander Ostrovskiy (a.ostrovskiy)
014 * @since 2.2.1.0
015 */
016public enum ResizeType implements Style {
017
018    NONE("none"),
019    BOTH("both"),
020    HORIZONTAL("horizontal"),
021    VERTICAL("vertical"),
022    INHERIT("inherit");
023
024    private String type;
025
026
027    ResizeType(String type) {
028        this.type = type;
029    }
030
031
032    @Override
033    public String get() {
034        return type;
035    }
036}
