016package com.github.gwtbootstrap.client.ui;
017
018import java.util.List;
019
020import com.github.gwtbootstrap.client.ui.base.DivWidget;
021import com.github.gwtbootstrap.client.ui.resources.Bootstrap;
022import com.google.gwt.user.client.ui.Widget;
023
024//@formatter:off
025/**
026 * The container for a tabbable nav.
027 * 
028 * @since 2.0.4.0
029 * @author Dominik Mayer
030 * @author ohashi keisuke
031 */
032//@formatter:on
033public class TabPanel extends DivWidget {
034
035    private static class TabContent extends DivWidget {
036
037        public TabContent() {
038            setStyleName(Bootstrap.tab_content);
039        }
040    }
041
042    private NavTabs tabs = new NavTabs();
043
044    private TabContent tabContent = new TabContent();
045
046    public TabPanel() {
047        this(Bootstrap.Tabs.ABOVE);
048    }
049
050    public TabPanel(Bootstrap.Tabs position) {
051        setStyle(position);
052    }
053
054    public void setTabPosition(String position) {
055        if(tabs.getParent() != null) {
056            remove(tabs);
057            remove(tabContent);
058        }
059        
060        if (position.equalsIgnoreCase("below")) {
061            setStyle(Bootstrap.Tabs.BELOW);
062            super.add(tabContent);
063            super.add(tabs);
064        } else if (position.equalsIgnoreCase("left")) {
065            setStyle(Bootstrap.Tabs.LEFT);
066            super.add(tabs);
067            super.add(tabContent);
068        } else if (position.equalsIgnoreCase("right")) {
069            setStyle(Bootstrap.Tabs.RIGHT);
070            super.add(tabs);
071            super.add(tabContent);
072        } else {
073            setStyle(Bootstrap.Tabs.ABOVE);
074            super.add(tabs);
075            super.add(tabContent);
076        }
077    }
078
079    @Override
080    public void add(Widget child) {
081
082        if (child instanceof TabPane) {
083            add((TabPane) child);
084            return;
085        }
086
087        if (child instanceof TabLink) {
088            add((TabLink) child);
089            return;
090        }
091        
092        if(child instanceof DropdownTab) {
093            add((DropdownTab) child);
094            return;
095        }
096        
097        throw new IllegalArgumentException("TabPanel can add only TabPane or TabLink or Tab or DorpdownTab");
098    }
099    
100    private void add(DropdownTab dropdownTab) {
101        
102        tabs.add(dropdownTab);
103        
104        List<Tab> tabList = dropdownTab.getTabList();
105        for (Tab tab : tabList) {
106            TabPane tabPane = tab.getTabPane();
107            tabContent.add(tabPane);
108        }
109    }
110
111    private void add(TabPane child) {
112
113        if(child.isCreateTabLink()) {
114            TabLink tabLink = new TabLink(child);
115            tabs.add(tabLink);
116        }
117        tabContent.add(child);
118    }
119    
120    private void add(final TabLink child) {
121        
122        if(child.isCreateTabPane() && child.getTabPane() == null){
123            TabPane pane = new TabPane(child.getText());
124            child.setTablePane(pane);
125            tabContent.add(pane);
126        } else if(child.getTabPane() != null) {
127            tabContent.add(child.getTabPane());
128        }
129        tabs.add(child);
130    }
131}
