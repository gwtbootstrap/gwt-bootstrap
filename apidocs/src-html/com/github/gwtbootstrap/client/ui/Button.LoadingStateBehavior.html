093    /**
094     * Button with optional icon.
095     * 
096     * <p>
097     * <h3>UiBinder Usage:</h3>
098     * 
099     * <pre>
100     * {@code <b:Button icon="TRASH" type="ERROR" toggle="true" loadingText="I'm loading..." completeText="Oh hoh, I completed the action!">Delete</b:Button>}
101     * </pre>
102     * 
103     * All arguments are optional.
104     * </p>
105     * 
106     * @since 2.0.3.0
107     * 
108     * @author Carlos Alexandro Becker
109     * 
110     * @author Dominik Mayer
111     * 
112     * @see <a
113     *      href="http://twitter.github.com/bootstrap/base-css.html#buttons">Bootstrap
114     *      documentation</a>
115     * @see ButtonGroup
116     * @see ButtonToolbar
117     * @see DropdownButton
118     * @see SplitDropdownButton
119     * @see NavbarButton
120     */
121    // @formatter:on
122    public class Button extends IconAnchor implements HasClickHandlers,
123                    HasDoubleClickHandlers, HasEnabled, HasType<ButtonType>,
124                    HasAllDragAndDropHandlers, HasAllFocusHandlers, HasAllGestureHandlers,
125                    HasAllKeyHandlers, HasAllMouseHandlers, HasAllTouchHandlers {
126    
127            private final LoadingStateBehavior state = new LoadingStateBehavior();
128    
129            /**
130             * Creates an empty Button.
131             */
132            public Button() {
133                    super();
134                    addStyleName(Constants.BTN);
135            }
136    
137            /**
138             * Creates a Button with the given caption.
139             * 
140             * @param caption
141             *            the caption of the Button
142             */
143            public Button(String caption) {
144                    this();
145                    setText(caption);
146            }
147    
148            /**
149             * Creates a Button with the given caption and icon.
150             * 
151             * @param caption
152             *            the caption of the Button
153             * @param icon
154             *            the Icon on the caption's left
155             */
156            public Button(String caption, IconType icon) {
157                    this(caption);
158                    setIcon(icon);
159            }
160    
161            /**
162             * Sets the type of the Button.
163             * <p>
164             * Different types give the button a different look.
165             * 
166             * @param type
167             *            the type of the Button.
168             */
169            public void setType(ButtonType type) {
170                    StyleHelper.changeStyle(this, type, ButtonType.class);
171            }
172    
173            /**
174             * Sets the size of the Button.
175             * 
176             * @param size
177             *            the size of the Button.
178             */
179            public void setSize(ButtonSize size) {
180                    StyleHelper.changeStyle(this, size, ButtonSize.class);
181            }
182    
183            /**
184             * Whether the Button is enabled or disabled.
185             * <p>
186             * A disabled widget cannot be used.
187             * 
188             * @return <code>false</code> if the Button is disabled.
189             */
190            public boolean isEnabled() {
191                    return !getStyleName().contains(Constants.DISABLED);
192            }
193    
194            /**
195             * Sets whether the Button is enabled or disabled.
196             * <p>
197             * A disabled widget cannot be used.
198             * 
199             * @param enabled
200             *            <code>false</code> if the Button should be disabled. Default:
201             *            <code>true</code>
202             */
203            public void setEnabled(boolean enabled) {
204                    if (enabled)
205                            removeStyleName(Constants.DISABLED);
206                    else
207                            addStyleName(Constants.DISABLED);
208            }
209    
210            /**
211             * Enable ou disable the data-toggle behavior.
212             * 
213             * @param toggle
214             *            <code>true</code> will enable this behavior.
215             *            <code>false</code> will disable it or do nothing if it never
216             *            was enabled.
217             */
218            public void setToggle(boolean toggle) {
219                    if (toggle)
220                            getElement().setAttribute(Constants.DATA_TOGGLE, "button");
221                    else
222                            getElement().removeAttribute(Constants.DATA_TOGGLE);
223            }
224    
225            /**
226             * Set a Loading Text to show when some action are in work with this button.
227             * 
228             * @see LoadingStateBehavior
229             * @param text
230             */
231            public void setLoadingText(String text) {
232                    if (text == null || text.trim().isEmpty()) {
233                            getElement().removeAttribute(Constants.DATA_LOADING_TEXT);
234                            return;
235                    }
236    
237                    getElement().setAttribute(Constants.DATA_LOADING_TEXT, text);
238            }
239    
240            /**
241             * Set a Loading Text to show when some action are completed with this
242             * button.
243             * 
244             * @see LoadingStateBehavior
245             * @param text
246             */
247            public void setCompleteText(String text) {
248                    if (text == null || text.trim().isEmpty()) {
249                            getElement().removeAttribute(Constants.DATA_COMPLETE_TEXT);
250                            return;
251                    }
252    
253                    getElement().setAttribute(Constants.DATA_COMPLETE_TEXT, text);
254            }
255    
256            /**
257             * A simple DSL to change the button state to loading, complete, or reset
258             * it.
259             * 
260             * @return
261             */
262            public LoadingStateBehavior state() {
263                    return state;
264            }
265    
266            /**
267             * A simple class to encapsulate the button state managing from the user.
268             * 
269             * @author Carlos Alexandro Becker
270             */
271            public class LoadingStateBehavior {
272    
273                    /**
274                     * put the button in the loading state.
275                     */
276                    public void loading() {
277                            setLoadingBehavior("loading");
278                    }
279    
280                    /**
281                     * reset the button state.
282                     */
283                    public void reset() {
284                            setLoadingBehavior("reset");
285                    }
286    
287                    /**
288                     * put the button in the completed state.
289                     */
290                    public void complete() {
291                            setLoadingBehavior("complete");
292                    }
293    
294                    private void setLoadingBehavior(String behavior) {
295                            setLoadingBehavior(getElement(), behavior);
296                    }
297    
298                    private native void setLoadingBehavior(Element e, String behavior) /*-{
299                            $wnd.jQuery(e).button(behavior);
300                    }-*/;
301            }
