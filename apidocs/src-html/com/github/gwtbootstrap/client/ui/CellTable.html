001    package com.github.gwtbootstrap.client.ui;
002    
003    import com.github.gwtbootstrap.client.ui.base.StyleHelper;
004    import com.google.gwt.core.client.GWT;
005    import com.google.gwt.resources.client.CssResource.ImportedWithPrefix;
006    import com.google.gwt.resources.client.ImageResource;
007    import com.google.gwt.user.client.ui.Image;
008    import com.google.gwt.user.client.ui.Widget;
009    import com.google.gwt.view.client.ProvidesKey;
010    
011    public class CellTable<T> extends com.google.gwt.user.cellview.client.CellTable<T> {
012    
013            public static enum TableType implements com.github.gwtbootstrap.client.ui.base.Style {
014                    BORDERED("table-bordered"),
015                    STRIPED("table-striped"),
016                    CONDENSED("table-condensed");
017                    
018                    private final String styleName;
019    
020                    private TableType(String styleName) {
021                            this.styleName = styleName;
022                    }
023    
024                    @Override
025                    public String get() {
026                            return styleName;
027                    }
028            }
029            
030            /**
031             * The default page size.
032             */
033            private static final int DEFAULT_PAGESIZE = 15;
034    
035            private static Resources DEFAULT_RESOURCES = null;
036    
037            private static Resources getDefaultResources() {
038                    if (DEFAULT_RESOURCES == null) {
039                            DEFAULT_RESOURCES = GWT.create(Resources.class);
040                    }
041                    return DEFAULT_RESOURCES;
042            }
043    
044            public interface Resources extends com.google.gwt.user.cellview.client.CellTable.Resources {
045    
046                    @Override
047                    @Source(Style.DEFAULT_CSS)
048                    public Style cellTableStyle();
049            }
050    
051            @ImportedWithPrefix("gwt-bootstrap-cellTable")
052            public interface Style extends com.google.gwt.user.cellview.client.CellTable.Style {
053    
054                    String DEFAULT_CSS = "com/github/gwtbootstrap/client/ui/GwtBootstrapCellTable.css";
055            }
056    
057            public CellTable() {
058                    this(DEFAULT_PAGESIZE);
059            }
060    
061            public CellTable(int pageSize, ProvidesKey<T> keyProvider) {
062                    this(pageSize, keyProvider , createDefaultLoadingIndicator(getDefaultResources()));
063            }
064    
065            public CellTable(int pageSize, ProvidesKey<T> keyProvider, Widget loadingIndicator) {
066                    super(pageSize, getDefaultResources(), keyProvider, loadingIndicator);
067                    setStyleName("table");
068            }
069    
070            public CellTable(int pageSize) {
071                    this(pageSize, null);
072            }
073    
074            public CellTable(ProvidesKey<T> keyProvider) {
075                    this(DEFAULT_PAGESIZE, keyProvider);
076            }
077            
078            public void setStriped(boolean striped) {
079                    if(striped) {
080                            StyleHelper.addStyle(this, TableType.STRIPED);
081                    } else {
082                            StyleHelper.removeStyle(this, TableType.STRIPED);
083                    }
084            }
085            
086            public void setBordered(boolean bordered) {
087                    if(bordered) {
088                            StyleHelper.addStyle(this, TableType.BORDERED);
089                    } else {
090                            StyleHelper.removeStyle(this, TableType.BORDERED);
091                    }
092            }
093    
094            public void setCondensed(boolean condensed) {
095                    if(condensed) {
096                            StyleHelper.addStyle(this, TableType.CONDENSED);
097                    } else {
098                            StyleHelper.removeStyle(this, TableType.CONDENSED);
099                    }
100            }
101            /**
102             * Create the default loading indicator using the loading image in the
103             * specified {@link Resources}.
104             * 
105             * @param resources
106             *            the resources
107             * @return a widget loading indicator
108             */
109            private static Widget createDefaultLoadingIndicator(Resources resources) {
110                    ImageResource loadingImg = resources.cellTableLoading();
111                    return (loadingImg == null)
112                                                                            ? null
113                                                                            : new Image(loadingImg);
114            }
115    }
