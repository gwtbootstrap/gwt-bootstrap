890        public enum Color {
891    
892            WHITE, BLACK;
893        }
894        
895        private static final String PREFIX = "icon-";
896        
897        private String type = "";
898        
899        private String color = "";
900        
901        public Icon() {
902            setElement(DOM.createElement("i"));
903        }
904    
905        public Icon(Type type, Color color) {
906            this(type);
907            setColor(color);
908        }
909        
910        public Icon(Type type) {
911            this(type.getStyle());
912        }
913        
914        public @UiConstructor Icon(String type) {
915            this();
916            setType(type);
917        }
918        
919        public void setType(Type type) {
920            setType(type.getStyle());
921        }
922        
923        public void setType(String type) {
924            this.type = PREFIX + type;
925            setStyles();
926        }
927        
928        public void setColor(String color) {
929            this.color = PREFIX + color;
930            setStyles();
931        }
932    
933            public void setColor(Color color) {
934            if (color == Color.WHITE) {
935                setColor("white");
936            } else
937                    setColor("");
938            }
939            
940            private void setStyles() {
941                    setStyleName("");
942                    if (!type.isEmpty())
943                            addStyleName(type);
944                    if (!color.isEmpty())
945                            addStyleName(color);
946            }
