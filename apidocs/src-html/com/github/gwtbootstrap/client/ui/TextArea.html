001    package com.github.gwtbootstrap.client.ui;
002    
003    import com.github.gwtbootstrap.client.ui.base.HasAlternateSize;
004    import com.github.gwtbootstrap.client.ui.base.HasId;
005    import com.github.gwtbootstrap.client.ui.base.HasPlaceholder;
006    import com.github.gwtbootstrap.client.ui.base.HasSize;
007    import com.github.gwtbootstrap.client.ui.base.PlaceholderHelper;
008    import com.github.gwtbootstrap.client.ui.base.SizeHelper;
009    import com.github.gwtbootstrap.client.ui.base.StyleHelper;
010    import com.github.gwtbootstrap.client.ui.constants.AlternateSize;
011    import com.github.gwtbootstrap.client.ui.constants.Constants;
012    import com.google.gwt.core.client.GWT;
013    import com.google.gwt.dom.client.Document;
014    import com.google.gwt.dom.client.Element;
015    
016    /**
017     * A TextArea for Bootstrap form.
018     * 
019     * @since 2.0.3.0
020     * @author ohashi keisuke
021     */
022    public class TextArea extends com.google.gwt.user.client.ui.TextArea implements HasSize , HasAlternateSize, HasPlaceholder, HasId {
023    
024            /** placeholderHelper */
025            private PlaceholderHelper placeholderHelper = GWT.create(PlaceholderHelper.class);
026    
027            /**
028             * Creates an empty text area.
029             */
030            public TextArea() {
031                    super(Document.get().createTextAreaElement());
032            }
033    
034            /**
035             * This constructor may be used by subclasses to explicitly use an existing
036             * element. This element must be a &lt;textarea&gt; element.
037             * 
038             * @param element
039             *            the element to be used
040             */
041            protected TextArea(Element element) {
042                    super(element.<Element> cast());
043            }
044    
045            /**
046             * {@inheritDoc}
047             */
048            @Override
049            public void setAlternateSize(AlternateSize size) {
050                    StyleHelper.changeStyle(this, size, AlternateSize.class);
051            }
052    
053            /**
054             * {@inheritDoc}
055             */
056            @Override
057            public void setSize(int size) {
058                    SizeHelper.setSize(this, size);
059            }
060    
061            /**
062             * {@inheritDoc}
063             */
064            @Override
065            public void setPlaceholder(String placeholder) {
066                    placeholderHelper.setPlaceholer(getElement(), placeholder);
067            }
068    
069            /**
070             * {@inheritDoc}
071             */
072            @Override
073            public String getPlaceholder() {
074                    return placeholderHelper.getPlaceholder(getElement());
075            }
076            
077            /**
078             * {@inheritDoc}
079             */
080            @Override
081            public String getId() {
082                    return getElement().getId();
083            }
084    
085            /**
086             * {@inheritDoc}
087             */
088            @Override
089            public void setId(String id) {
090                    getElement().setId(id);
091            }
092            
093            /**
094             * {@inheritDoc}
095             */
096            @Override
097            public void setEnabled(boolean enabled) {
098                    super.setEnabled(enabled);
099                    if(enabled) {
100                            removeStyleName(Constants.DISABLED);
101                    } else {
102                            addStyleName(Constants.DISABLED);
103                    }
104            }
105    
106    
107    }
