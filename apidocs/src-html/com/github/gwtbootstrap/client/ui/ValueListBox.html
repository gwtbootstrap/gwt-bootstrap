009    import com.github.gwtbootstrap.client.ui.base.HasAlternateSize;
010    import com.github.gwtbootstrap.client.ui.base.HasId;

033    /**
034     * The TW Bootstrap style ValueListBox
035     * 
036     * @param <T> the value type
037     * @author ohashi keisuke
038     * @since 2.0.3.0
039     */
040    public class ValueListBox<T> extends Composite implements HasConstrainedValue<T>, IsEditor<TakesValueEditor<T>>, HasName, HasId, HasEnabled, HasSize, HasAlternateSize, IsSearchQuery {
041    
042            private final Map<Object, Integer> valueKeyToIndex = new HashMap<Object, Integer>();
043    
044            private final List<T> valueList = new ArrayList<T>();
045    
046            private final Renderer<T> renderer;
047    
048            private final ProvidesKey<T> keyProvider;
049    
050            private TakesValueEditor<T> editor;
051    
052            private T value;
053    
054            public ValueListBox(Renderer<T> renderer) {
055                    this(renderer, new SimpleKeyProvider<T>());
056            }
057    
058            public ValueListBox(Renderer<T> renderer,
059                    ProvidesKey<T> keyProvider) {
060                    this.keyProvider = keyProvider;
061                    this.renderer = renderer;
062                    initWidget(new ListBox());
063    
064                    getListBox().addChangeHandler(new ChangeHandler() {
065    
066                            public void onChange(ChangeEvent event) {
067                                    int selectedIndex = getListBox().getSelectedIndex();
068    
069                                    if (selectedIndex < 0) {
070                                            return;
071                                    }
072                                    T newValue = valueList.get(selectedIndex);
073                                    setValue(newValue, true);
074                            }
075                    });
076            }
077    
078            public void setName(String name) {
079                    getListBox().setName(name);
080            }
081    
082            public String getName() {
083                    return getListBox().getName();
084            }
085    
086            public HandlerRegistration addValueChangeHandler(ValueChangeHandler<T> handler) {
087                    return addHandler(handler, ValueChangeEvent.getType());
088            }
089    
090            /**
091             * Returns a {@link TakesValueEditor} backed by the ValueListBox.
092             */
093            public TakesValueEditor<T> asEditor() {
094                    if (editor == null) {
095                            editor = TakesValueEditor.of(this);
096                    }
097                    return editor;
098            }
099    
100            public T getValue() {
101                    return value;
102            }
103    
104            public void setAcceptableValues(Collection<T> newValues) {
105    
106                    valueList.clear();
107                    
108                    valueKeyToIndex.clear();
109                    
110                    ListBox listBox = getListBox();
111                    
112                    listBox.clear();
113    
114                    for (T nextNewValue : newValues) {
115                            addValue(nextNewValue);
116                    }
117    
118                    updateListBox();
119            }
120    
121            /**
122             * Set the value and display it in the select element. Add the value to the
123             * acceptable set if it is not already there.
124             */
125            public void setValue(T value) {
126                    setValue(value, false);
127            }
128    
129            public void setValue(T value, boolean fireEvents) {
130                    if (value == this.value || (this.value != null && this.value.equals(value))) {
131                            return;
132                    }
133    
134                    T before = this.value;
135                    this.value = value;
136                    updateListBox();
137    
138                    if (fireEvents) {
139                            ValueChangeEvent.fireIfNotEqual(this, before, value);
140                    }
141            }
142    
143            private void addValue(T value) {
144                    Object key = keyProvider.getKey(value);
145                    if (valueKeyToIndex.containsKey(key)) {
146                            throw new IllegalArgumentException("Duplicate value: " + value);
147                    }
148    
149                    valueKeyToIndex.put(key, valueList.size());
150                    valueList.add(value);
151                    getListBox().addItem(renderer.render(value));
152                    assert valueList.size() == getListBox().getItemCount();
153            }
154    
155            private ListBox getListBox() {
156                    return (ListBox) getWidget();
157            }
158    
159            private void updateListBox() {
160                    Object key = keyProvider.getKey(value);
161                    Integer index = valueKeyToIndex.get(key);
162                    if (index == null) {
163                            addValue(value);
164                    }
165    
166                    index = valueKeyToIndex.get(key);
167                    getListBox().setSelectedIndex(index);
168            }
169    
170            /**
171             * {@inheritDoc}
172             */
173            @Override
174            public String getId() {
175                    return getListBox().getId();
176            }
177    
178            /**
179             * {@inheritDoc}
180             */
181            @Override
182            public void setId(String id) {
183                    getListBox().setId(id);
184            }
185    
186            /**
187             * {@inheritDoc}
188             */
189            @Override
190            public void setSearchQuery(boolean searchQuery) {
191                    SearchQueryStyleHelper.setSearchQuery(this, searchQuery);
192            }
193    
194            /**
195             * {@inheritDoc}
196             */
197            @Override
198            public boolean isSearchQuery() {
199                    return SearchQueryStyleHelper.isSearchQuery(this);
200            }
201    
202    
203            /**
204             * {@inheritDoc}
205             */
206            @Override
207            public void setAlternateSize(AlternateSize size) {
208                    StyleHelper.changeStyle(this, size, AlternateSize.class);
209            }
210    
211            /**
212             * {@inheritDoc}
213             */
214            @Override
215            public void setSize(int size) {
216                    SizeHelper.setSize(this, size);
217            }
218            /**
219             * {@inheritDoc}
220             */
221            @Override
222            public void setEnabled(boolean enabled) {
223                    getListBox().setEnabled(enabled);
224                    if(enabled) {
225                            removeStyleName(Constants.DISABLED);
226                    } else {
227                            addStyleName(Constants.DISABLED);
228                    }
229            }
230    
231            @Override
232            public boolean isEnabled() {
233                    return getListBox().isEnabled();
234            }
235    }
