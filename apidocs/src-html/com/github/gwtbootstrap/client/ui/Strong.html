016    package com.github.gwtbootstrap.client.ui;
017    
018    import com.github.gwtbootstrap.client.ui.base.AbstractTypography;
019    import com.google.gwt.user.client.DOM;
020    
021    //@formatter:off
022    /**
023     * Simple wrapper for an HTML {@code <strong>} tag.
024     * 
025     * @since 2.0.3.0
026     * 
027     * @author Carlos Alexandro Becker
028     * 
029     * @see <a href="http://twitter.github.com/bootstrap/base-css.html#typography">Bootstrap documentation</a>
030     */
031    //@formatter:on
032    public class Strong extends AbstractTypography {
033    
034            /**
035             * Creates a new widget.
036             */
037            public Strong() {
038                    setElement(DOM.createElement("strong"));
039            }
040    
041            /**
042             * Creates a new widget with given text.
043             * 
044             * @param text
045             *            the text of the widget
046             */
047            public Strong(String text) {
048                    this();
049                    setText(text);
050            }
051    }
