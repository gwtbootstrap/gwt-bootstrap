016    package com.github.gwtbootstrap.client.ui;
017    
018    import com.github.gwtbootstrap.client.ui.base.DropdownBase;
019    import com.github.gwtbootstrap.client.ui.base.IconAnchor;
020    import com.github.gwtbootstrap.client.ui.constants.ButtonType;
021    import com.github.gwtbootstrap.client.ui.constants.IconType;
022    import com.github.gwtbootstrap.client.ui.resources.ButtonSize;
023    
024    //@formatter:off
025    /**
026     * Button with a dropdown menu.
027     * 
028     * <p>
029     * <h3>UiBinder Usage:</h3>
030     * 
031     * <pre>
032     * {@code
033     * <b:DropdownButton text="I am the Caption">
034     *     <b:NavHeader>Header</b:NavHeader>
035     *     <b:NavLink>Link 1</b:NavLink>
036     *     <b:NavLink>Link 2</b:NavLink>
037     * </b:DropdownButton>
038     * }
039     * </pre>
040     * </p>
041     * 
042     * @since 2.0.3.0
043     * 
044     * @author Dominik Mayer
045     * 
046     * @see <a href="http://twitter.github.com/bootstrap/components.html#buttonDropdowns">Bootstrap documentation</a>
047     * @see Dropdown
048     * @see SplitDropdownButton
049     */
050    //@formatter:on
051    public class DropdownButton extends DropdownBase {
052    
053            private Button trigger;
054    
055            /**
056             * Creates a DropdownButton without a caption.
057             */
058            public DropdownButton() {
059                    super("div");
060                    addStyleName("btn-group");
061            }
062    
063            /**
064             * Creates a DropdownButton with the given caption.
065             * 
066             * @param caption
067             *            the button's caption
068             */
069            public DropdownButton(String caption) {
070                    this();
071                    setText(caption);
072            }
073    
074            /**
075             * {@inheritDoc}
076             */
077            @Override
078            protected IconAnchor createTrigger() {
079                    trigger = new Button();
080                    trigger.setCaret(true);
081                    return trigger;
082            }
083    
084            /**
085             * Sets the button's size.
086             * 
087             * @param size
088             *            the button's size
089             */
090            public void setSize(ButtonSize size) {
091                    trigger.setSize(size);
092            }
093    
094            /**
095             * Sets the button's type.
096             * 
097             * @param type
098             *            the button's type
099             */
100            public void setType(ButtonType type) {
101                    trigger.setType(type);
102            }
103    
104            /**
105             * Sets the button's icon.
106             * 
107             * @param type
108             *            the icon's type
109             */
110            public void setIcon(IconType type) {
111                    trigger.setIcon(type);
112            }
113    
114    }
