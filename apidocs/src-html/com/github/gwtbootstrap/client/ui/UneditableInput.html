001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.HasAlternateSize;
004import com.github.gwtbootstrap.client.ui.base.HasId;
005import com.github.gwtbootstrap.client.ui.base.HasSize;
006import com.github.gwtbootstrap.client.ui.base.InlineLabel;
007import com.github.gwtbootstrap.client.ui.base.IsSearchQuery;
008import com.github.gwtbootstrap.client.ui.base.SearchQueryStyleHelper;
009import com.github.gwtbootstrap.client.ui.base.SizeHelper;
010import com.github.gwtbootstrap.client.ui.base.StyleHelper;
011import com.github.gwtbootstrap.client.ui.constants.AlternateSize;
012import com.github.gwtbootstrap.client.ui.constants.Constants;
013import com.google.gwt.dom.client.Element;
014import com.google.gwt.i18n.shared.DirectionEstimator;
015import com.google.gwt.user.client.ui.Label;
016
017
018
019/**
020 * Uneditable style text input.
021 * 
022 * @since 2.0.3.0
023 * @author ohashi keisuke
024 */
025public class UneditableInput extends InlineLabel implements HasAlternateSize, IsSearchQuery, HasSize, HasId {
026
027        /**
028         * Creates an empty widget.
029         */
030        public UneditableInput() {
031                super();
032                setStyleName(Constants.UNEDITABLE_INPUT);
033        }
034
035        /**
036         * This constructor may be used by subclasses to explicitly use an existing
037         * element. This element must be either a &lt;div&gt; &lt;span&gt; element.
038         * 
039         * @param element
040         *            the element to be used
041         */
042        protected UneditableInput(Element element) {
043                super(element);
044        }
045
046        /**
047         * Creates a label with the specified text and direction.
048         * 
049         * @param text
050         *            the new label's text
051         * @param dir
052         *            the text's direction. Note: {@code Direction.DEFAULT} means
053         *            direction should be inherited from the widget's parent
054         *            element.
055         */
056        public UneditableInput(String text, Direction dir) {
057                super(text, dir);
058        }
059
060        /**
061         * Creates a label with the specified text and a default direction
062         * estimator.
063         * 
064         * @param text
065         *            the new label's text
066         * @param directionEstimator
067         *            A DirectionEstimator object used for automatic direction
068         *            adjustment. For convenience,
069         *            {@link Label#DEFAULT_DIRECTION_ESTIMATOR} can be used.
070         */
071        public UneditableInput(String text, DirectionEstimator directionEstimator) {
072                super(text, directionEstimator);
073        }
074
075        /**
076         * Creates a widget with the specified text.
077         * 
078         * @param text
079         *            the new widget's text
080         */
081        public UneditableInput(String text) {
082                super(text);
083        }
084
085        /**
086         * {@inheritDoc}
087         */
088        @Override
089        public void setSearchQuery(boolean searchQuery) {
090                SearchQueryStyleHelper.setSearchQuery(this, searchQuery);
091        }
092
093        /**
094         * {@inheritDoc}
095         */
096        @Override
097        public boolean isSearchQuery() {
098                return SearchQueryStyleHelper.isSearchQuery(this);
099        }
100
101        /**
102         * {@inheritDoc}
103         */
104        @Override
105        public void setAlternateSize(AlternateSize size) {
106                StyleHelper.changeStyle(this, size, AlternateSize.class);
107        }
108
109        /**
110         * {@inheritDoc}
111         */
112        @Override
113        public void setSize(int size) {
114                SizeHelper.setSize(this, size);
115        }
116        
117        /**
118         * {@inheritDoc}
119         */
120        @Override
121        public String getId() {
122                return getElement().getId();
123        }
124
125        /**
126         * {@inheritDoc}
127         */
128        @Override
129        public void setId(String id) {
130                getElement().setId(id);
131        }
132}
