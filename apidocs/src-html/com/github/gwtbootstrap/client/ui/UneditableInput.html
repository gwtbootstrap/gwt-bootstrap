016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.base.HasAlternateSize;
019import com.github.gwtbootstrap.client.ui.base.HasId;
020import com.github.gwtbootstrap.client.ui.base.HasSize;
021import com.github.gwtbootstrap.client.ui.base.InlineLabel;
022import com.github.gwtbootstrap.client.ui.base.IsSearchQuery;
023import com.github.gwtbootstrap.client.ui.base.SearchQueryStyleHelper;
024import com.github.gwtbootstrap.client.ui.base.SizeHelper;
025import com.github.gwtbootstrap.client.ui.base.StyleHelper;
026import com.github.gwtbootstrap.client.ui.constants.AlternateSize;
027import com.github.gwtbootstrap.client.ui.constants.Constants;
028import com.google.gwt.dom.client.Element;
029import com.google.gwt.i18n.shared.DirectionEstimator;
030import com.google.gwt.user.client.ui.Label;
031
032
033
034/**
035 * Uneditable style text input.
036 * 
037 * @since 2.0.4.0
038 * @author ohashi keisuke
039 */
040public class UneditableInput extends InlineLabel implements HasAlternateSize, IsSearchQuery, HasSize, HasId {
041
042        /**
043         * Creates an empty widget.
044         */
045        public UneditableInput() {
046                super();
047                setStyleName(Constants.UNEDITABLE_INPUT);
048        }
049
050        /**
051         * This constructor may be used by subclasses to explicitly use an existing
052         * element. This element must be either a &lt;div&gt; &lt;span&gt; element.
053         * 
054         * @param element
055         *            the element to be used
056         */
057        protected UneditableInput(Element element) {
058                super(element);
059        }
060
061        /**
062         * Creates a label with the specified text and direction.
063         * 
064         * @param text
065         *            the new label's text
066         * @param dir
067         *            the text's direction. Note: {@code Direction.DEFAULT} means
068         *            direction should be inherited from the widget's parent
069         *            element.
070         */
071        public UneditableInput(String text, Direction dir) {
072                super(text, dir);
073        }
074
075        /**
076         * Creates a label with the specified text and a default direction
077         * estimator.
078         * 
079         * @param text
080         *            the new label's text
081         * @param directionEstimator
082         *            A DirectionEstimator object used for automatic direction
083         *            adjustment. For convenience,
084         *            {@link Label#DEFAULT_DIRECTION_ESTIMATOR} can be used.
085         */
086        public UneditableInput(String text, DirectionEstimator directionEstimator) {
087                super(text, directionEstimator);
088        }
089
090        /**
091         * Creates a widget with the specified text.
092         * 
093         * @param text
094         *            the new widget's text
095         */
096        public UneditableInput(String text) {
097                super(text);
098        }
099
100        /**
101         * {@inheritDoc}
102         */
103        @Override
104        public void setSearchQuery(boolean searchQuery) {
105                SearchQueryStyleHelper.setSearchQuery(this, searchQuery);
106        }
107
108        /**
109         * {@inheritDoc}
110         */
111        @Override
112        public boolean isSearchQuery() {
113                return SearchQueryStyleHelper.isSearchQuery(this);
114        }
115
116        /**
117         * {@inheritDoc}
118         */
119        @Override
120        public void setAlternateSize(AlternateSize size) {
121                StyleHelper.changeStyle(this, size, AlternateSize.class);
122        }
123
124        /**
125         * {@inheritDoc}
126         */
127        @Override
128        public void setSize(int size) {
129                SizeHelper.setSize(this, size);
130        }
131        
132        /**
133         * {@inheritDoc}
134         */
135        @Override
136        public String getId() {
137                return getElement().getId();
138        }
139
140        /**
141         * {@inheritDoc}
142         */
143        @Override
144        public void setId(String id) {
145                getElement().setId(id);
146        }
147}
