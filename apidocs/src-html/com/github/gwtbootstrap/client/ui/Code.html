016    package com.github.gwtbootstrap.client.ui;
017    
018    import com.github.gwtbootstrap.client.ui.base.AbstractTypography;
019    import com.github.gwtbootstrap.client.ui.resources.prettify.HasProgrammingLanguage;
020    import com.github.gwtbootstrap.client.ui.resources.prettify.PrettifyHelper;
021    import com.google.gwt.user.client.DOM;
022    
023    /**
024     * Simple inline code.
025     * 
026     * @author Dominik Mayer
027     * @author Carlos A Becker
028     * 
029     */
030    public class Code extends AbstractTypography implements HasProgrammingLanguage {
031    
032            private final PrettifyHelper helper;
033            
034            public Code() {
035                    setElement(DOM.createElement("code"));
036                    helper = new PrettifyHelper(this);
037            }
038            
039            @Override
040            protected void onLoad() {
041                    super.onLoad();
042                    helper.configure();
043            }
044    
045            public void setLang(String lang) {
046                    helper.setLang(lang);
047            }
048    }
