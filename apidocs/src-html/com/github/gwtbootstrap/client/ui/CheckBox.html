174            protected CheckBox(Element elem) {
175                    super(DOM.createLabel());
176                    
177                    assert elem.hasAttribute("type") : "The elem should has type attributes";
178    
179                    //TODO 2012/05/06 ohashi keisuke. ugly code......
180                    if(Constants.CHECKBOX.toLowerCase().equals(elem.getAttribute("type").toLowerCase())) {
181                            this.setStyleName(Constants.CHECKBOX);
182                    } else if(Constants.RADIO.toLowerCase().equals(elem.getAttribute("type").toLowerCase())){
183                            this.setStyleName(Constants.RADIO);
184                    }
185    
186                    inputElem = InputElement.as(elem);
187                    spanElem = Document.get().createSpanElement();
188                    
189                    getElement().appendChild(inputElem);
190                    getElement().appendChild(spanElem);
191    
192                    String uid = DOM.createUniqueId();
193                    inputElem.setPropertyString("id", uid);
194                    asLabel().setHtmlFor(uid);
195                    directionalTextHelper = new DirectionalTextHelper(spanElem, false);
196                    setTabIndex(0);
197            }

199            public HandlerRegistration addValueChangeHandler(ValueChangeHandler<Boolean> handler) {
200                    // Is this the first value change handler? If so, time to add handlers
201                    if (!valueChangeHandlerInitialized) {
202                            ensureDomEventHandlers();
203                            valueChangeHandlerInitialized = true;
204                    }
205                    return addHandler(handler, ValueChangeEvent.getType());
206            }

417            /**
418             * Checks or unchecks the check box, firing {@link ValueChangeEvent} if
419             * appropriate.
420             * <p>
421             * Note that this <em>does not</em> set the value property of the checkbox
422             * input element wrapped by this widget. For access to that property, see
423             * {@link #setFormValue(String)}
424             * 
425             * @param value
426             *            true to check, false to uncheck; null value implies false
427             * @param fireEvents
428             *            If true, and value has changed, fire a
429             *            {@link ValueChangeEvent}
430             */
431            public void setValue(Boolean value, boolean fireEvents) {
432                    if (value == null) {
433                            value = Boolean.FALSE;
434                    }
435    
436                    Boolean oldValue = getValue();
437                    inputElem.setChecked(value);
438                    inputElem.setDefaultChecked(value);
439                    if (value.equals(oldValue)) {
440                            return;
441                    }
442                    if (fireEvents) {
443                            ValueChangeEvent.fire(this, value);
444                    }
445            }

453            // Unlike other widgets the CheckBox sinks on its inputElement, not
454            // its wrapper
455            @Override
456            public void sinkEvents(int eventBitsToAdd) {
457                    if (isOrWasAttached()) {
458                            Event.sinkEvents(inputElem, eventBitsToAdd | Event.getEventsSunk(inputElem));
459                    } else {
460                            super.sinkEvents(eventBitsToAdd);
461                    }
462            }

464            protected void ensureDomEventHandlers() {
465                    addClickHandler(new ClickHandler() {
466    
467                            public void onClick(ClickEvent event) {
468                                    // Checkboxes always toggle their value, no need to compare
469                                    // with old value. Radio buttons are not so lucky, see
470                                    // overrides in RadioButton
471                                    ValueChangeEvent.fire(CheckBox.this, getValue());
472                            }
473                    });
474            }

503            /**
504             * This method is called when a widget is detached from the browser's
505             * document. Overridden because of IE bug that throws away checked state and
506             * in order to clear the event listener off of the <code>inputElem</code>.
507             */
508            @Override
509            protected void onUnload() {
510                    // Clear out the inputElem's event listener (breaking the circular
511                    // reference between it and the widget).
512                    setEventListener(asOld(inputElem), null);
513                    setValue(getValue());
514            }

516            /**
517             * Replace the current input element with a new one. Preserves all state
518             * except for the name property, for nasty reasons related to radio button
519             * grouping. (See implementation of {@link RadioButton#setName}.)
520             * 
521             * @param elem
522             *            the new input element
523             */
524            protected void replaceInputElement(Element elem) {
525                    InputElement newInputElem = InputElement.as(elem);
526                    // Collect information we need to set
527                    int tabIndex = getTabIndex();
528                    boolean checked = getValue();
529                    boolean enabled = isEnabled();
530                    String formValue = getFormValue();
531                    String uid = inputElem.getId();
532                    String accessKey = inputElem.getAccessKey();
533                    int sunkEvents = Event.getEventsSunk(inputElem);
534    
535                    // Clear out the old input element
536                    setEventListener(asOld(inputElem), null);
537    
538                    getElement().replaceChild(newInputElem, inputElem);
539    
540                    // Sink events on the new element
541                    Event.sinkEvents(elem, Event.getEventsSunk(inputElem));
542                    Event.sinkEvents(inputElem, 0);
543                    inputElem = newInputElem;
544    
545                    // Setup the new element
546                    Event.sinkEvents(inputElem, sunkEvents);
547                    inputElem.setId(uid);
548                    if (!"".equals(accessKey)) {
549                            inputElem.setAccessKey(accessKey);
550                    }
551                    setTabIndex(tabIndex);
552                    setValue(checked);
553                    setEnabled(enabled);
554                    setFormValue(formValue);
555    
556                    // Set the event listener
557                    if (isAttached()) {
558                            setEventListener(asOld(inputElem), this);
559                    }
560            }
561    
562            private Element asOld(com.google.gwt.dom.client.Element elem) {
563                    Element oldSchool = elem.cast();
564                    return oldSchool;
565            }
566    
567            private void setEventListener(com.google.gwt.dom.client.Element e, EventListener listener) {
568                    DOM.setEventListener(asOld(e), listener);
569            }
