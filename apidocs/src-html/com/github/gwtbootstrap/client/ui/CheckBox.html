189        protected CheckBox(Element elem) {
190                super(DOM.createLabel());
191                
192                assert elem.hasAttribute("type") : "The elem should has type attributes";
193
194                //TODO 2012/05/06 ohashi keisuke. ugly code......
195                if(Constants.CHECKBOX.toLowerCase().equals(elem.getAttribute("type").toLowerCase())) {
196                        this.setStyleName(Constants.CHECKBOX);
197                } else if(Constants.RADIO.toLowerCase().equals(elem.getAttribute("type").toLowerCase())){
198                        this.setStyleName(Constants.RADIO);
199                }
200
201                inputElem = InputElement.as(elem);
202                spanElem = Document.get().createSpanElement();
203                
204                getElement().appendChild(inputElem);
205                getElement().appendChild(spanElem);
206
207                String uid = DOM.createUniqueId();
208                inputElem.setPropertyString("id", uid);
209                asLabel().setHtmlFor(uid);
210                directionalTextHelper = new DirectionalTextHelper(spanElem, false);
211                setTabIndex(0);
212        }

214        public HandlerRegistration addValueChangeHandler(ValueChangeHandler<Boolean> handler) {
215                // Is this the first value change handler? If so, time to add handlers
216                if (!valueChangeHandlerInitialized) {
217                        ensureDomEventHandlers();
218                        valueChangeHandlerInitialized = true;
219                }
220                return addHandler(handler, ValueChangeEvent.getType());
221        }

432        /**
433         * Checks or unchecks the check box, firing {@link ValueChangeEvent} if
434         * appropriate.
435         * <p>
436         * Note that this <em>does not</em> set the value property of the checkbox
437         * input element wrapped by this widget. For access to that property, see
438         * {@link #setFormValue(String)}
439         * 
440         * @param value
441         *            true to check, false to uncheck; null value implies false
442         * @param fireEvents
443         *            If true, and value has changed, fire a
444         *            {@link ValueChangeEvent}
445         */
446        public void setValue(Boolean value, boolean fireEvents) {
447                if (value == null) {
448                        value = Boolean.FALSE;
449                }
450
451                Boolean oldValue = getValue();
452                inputElem.setChecked(value);
453                inputElem.setDefaultChecked(value);
454                if (value.equals(oldValue)) {
455                        return;
456                }
457                if (fireEvents) {
458                        ValueChangeEvent.fire(this, value);
459                }
460        }

468        // Unlike other widgets the CheckBox sinks on its inputElement, not
469        // its wrapper
470        @Override
471        public void sinkEvents(int eventBitsToAdd) {
472                if (isOrWasAttached()) {
473                        Event.sinkEvents(inputElem, eventBitsToAdd | Event.getEventsSunk(inputElem));
474                } else {
475                        super.sinkEvents(eventBitsToAdd);
476                }
477        }

479        protected void ensureDomEventHandlers() {
480                addClickHandler(new ClickHandler() {
481
482                        public void onClick(ClickEvent event) {
483                                // Checkboxes always toggle their value, no need to compare
484                                // with old value. Radio buttons are not so lucky, see
485                                // overrides in RadioButton
486                                ValueChangeEvent.fire(CheckBox.this, getValue());
487                        }
488                });
489        }

518        /**
519         * This method is called when a widget is detached from the browser's
520         * document. Overridden because of IE bug that throws away checked state and
521         * in order to clear the event listener off of the <code>inputElem</code>.
522         */
523        @Override
524        protected void onUnload() {
525                // Clear out the inputElem's event listener (breaking the circular
526                // reference between it and the widget).
527                setEventListener(asOld(inputElem), null);
528                setValue(getValue());
529        }

531        /**
532         * Replace the current input element with a new one. Preserves all state
533         * except for the name property, for nasty reasons related to radio button
534         * grouping. (See implementation of {@link RadioButton#setName}.)
535         * 
536         * @param elem
537         *            the new input element
538         */
539        protected void replaceInputElement(Element elem) {
540                InputElement newInputElem = InputElement.as(elem);
541                // Collect information we need to set
542                int tabIndex = getTabIndex();
543                boolean checked = getValue();
544                boolean enabled = isEnabled();
545                String formValue = getFormValue();
546                String uid = inputElem.getId();
547                String accessKey = inputElem.getAccessKey();
548                int sunkEvents = Event.getEventsSunk(inputElem);
549
550                // Clear out the old input element
551                setEventListener(asOld(inputElem), null);
552
553                getElement().replaceChild(newInputElem, inputElem);
554
555                // Sink events on the new element
556                Event.sinkEvents(elem, Event.getEventsSunk(inputElem));
557                Event.sinkEvents(inputElem, 0);
558                inputElem = newInputElem;
559
560                // Setup the new element
561                Event.sinkEvents(inputElem, sunkEvents);
562                inputElem.setId(uid);
563                if (!"".equals(accessKey)) {
564                        inputElem.setAccessKey(accessKey);
565                }
566                setTabIndex(tabIndex);
567                setValue(checked);
568                setEnabled(enabled);
569                setFormValue(formValue);
570
571                // Set the event listener
572                if (isAttached()) {
573                        setEventListener(asOld(inputElem), this);
574                }
575        }
576
577        private Element asOld(com.google.gwt.dom.client.Element elem) {
578                Element oldSchool = elem.cast();
579                return oldSchool;
580        }
581
582        private void setEventListener(com.google.gwt.dom.client.Element e, EventListener listener) {
583                DOM.setEventListener(asOld(e), listener);
584        }
