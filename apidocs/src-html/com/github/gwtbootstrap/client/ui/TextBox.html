001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.constants.Constants;
004
005/**
006 * A TextBox with a placeholder attribute.
007 * 
008 * @since 2.0.2.0
009 * 
010 * @author Carlos Alexandro Becker
011 * 
012 */
013public class TextBox extends com.github.gwtbootstrap.client.ui.base.TextBox {
014
015        /**
016         * Set a placeholder attribute (HTML5).
017         * 
018         * @param placeholder
019         *            the String to show. Use a empty string or null to remove the attribute.
020         */
021        public void setPlaceholder(String placeholder) {
022                if (placeholder == null || placeholder.trim().isEmpty()) {
023                        getElement().removeAttribute(Constants.PLACEHOLDER);
024                        return;
025                }
026                getElement().setAttribute(Constants.PLACEHOLDER, placeholder);
027        }
028
029}
