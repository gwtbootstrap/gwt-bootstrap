001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.HasAlternateSize;
004import com.github.gwtbootstrap.client.ui.base.HasPlaceholder;
005import com.github.gwtbootstrap.client.ui.base.HasSize;
006import com.github.gwtbootstrap.client.ui.base.IsSearchQuery;
007import com.github.gwtbootstrap.client.ui.base.PlaceholderHelper;
008import com.github.gwtbootstrap.client.ui.base.SearchQueryStyleHelper;
009import com.github.gwtbootstrap.client.ui.base.SizeHelper;
010import com.github.gwtbootstrap.client.ui.base.StyleHelper;
011import com.github.gwtbootstrap.client.ui.constants.AlternateSize;
012import com.google.gwt.core.client.GWT;
013
014/**
015 * A TextBox for Bootstrap form.
016 * 
017 * @since 2.0.3.0
018 * 
019 * @author Carlos Alexandro Becker
020 * @author ohashi keisuke
021 * 
022 */
023public class TextBox extends com.github.gwtbootstrap.client.ui.base.TextBox implements HasPlaceholder, HasAlternateSize, IsSearchQuery, HasSize {
024
025        /** placeholderHelper */
026        private PlaceholderHelper placeholderHelper = GWT.create(PlaceholderHelper.class);
027
028        /**
029         * {@inheritDoc}
030         */
031        @Override
032        public void setPlaceholder(String placeholder) {
033                placeholderHelper.setPlaceholer(getElement(), placeholder);
034        }
035
036        /**
037         * {@inheritDoc}
038         */
039        @Override
040        public String getPlaceholder() {
041                return placeholderHelper.getPlaceholder(getElement());
042        }
043
044        /**
045         * {@inheritDoc}
046         */
047        @Override
048        public void setSearchQuery(boolean searchQuery) {
049                SearchQueryStyleHelper.setSearchQuery(this, searchQuery);
050        }
051
052        /**
053         * {@inheritDoc}
054         */
055        @Override
056        public boolean isSearchQuery() {
057                return SearchQueryStyleHelper.isSearchQuery(this);
058        }
059
060        /**
061         * {@inheritDoc}
062         */
063        @Override
064        public void setAlternateSize(AlternateSize size) {
065                StyleHelper.changeStyle(this, size, AlternateSize.class);
066        }
067
068        /**
069         * {@inheritDoc}
070         */
071        @Override
072        public void setSize(int size) {
073                SizeHelper.setSize(this, size);
074        }
075
076        // TODO 2012/05/05 ohashi keisuke. Should create setter for uneditable,disable
077}
