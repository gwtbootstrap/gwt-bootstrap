037    /**
038     * Popup dialog with optional header and {@link ModalFooter footer.}
039     * <p>
040     * By default, all other Modals are closed once a new one is opened. This
041     * setting can be {@link #setHideOthers(boolean) overridden.}
042     * 
043     * <p>
044     * <h3>UiBinder Usage:</h3>
045     * 
046     * <pre>
047     * {@code
048     * <b:Modal title="My Modal" backdrop="STATIC">
049     *     <g:Label>Modal Content!</g:Label>
050     *     <b:ModalFooter>
051     *         <b:Button icon="FILE">Save</b:Button>
052     *     </b:ModalFooter>
053     * </b:Modal>
054     * }
055     * </pre>
056     * All arguments are optional.
057     * </p>
058     * 
059     * @since 2.0.3.0
060     * 
061     * @author Carlos Alexandro Becker
062     * 
063     * @author Dominik Mayer
064     * 
065     * @see <a href="http://twitter.github.com/bootstrap/javascript.html#modals">Bootstrap documentation</a>
066     * @see PopupPanel
067     */
068    //@formatter:on
069    public class Modal extends DivWidget implements HasVisibility,
070                    HasVisibleHandlers, IsAnimated {
071    
072            private static Set<Modal> currentlyShown = new HashSet<Modal>();
073    
074            private final DivWidget header = new DivWidget();
075            private final DivWidget body = new DivWidget("modal-body");
076    
077            private boolean keyboard = true;
078            private BackdropType backdropType = BackdropType.NORMAL;
079            private boolean show = false;
080            private boolean hideOthers = true;
081    
082            private boolean configured = false;
083    
084            private Close close = new Close(DismissType.MODAL);
085            /**
086             * Creates an empty, hidden widget.
087             */
088            public Modal() {
089                    super("modal");
090                    super.add(header);
091                    super.add(body);
092                    setVisible(false);
093                    RootPanel.get().add(this);
094                    setHandlerFunctions(getElement());
095            }
096    
097            /**
098             * Creates an empty, hidden widget with specified show behavior.
099             * 
100             * @param animated
101             *            <code>true</code> if the widget should be animated.
102             * 
103             */
104            public Modal(boolean animated) {
105                this(animated, false);
106            }
107            
108            /**
109             * Creates an empty, hidden widget with specified show behavior.
110             * 
111             * @param animated
112             *            <code>true</code> if the widget should be animated.
113             * 
114             * @param dynamicSafe
115             *            <code>true</code> removes from RootPanel when hidden
116             */
117            public Modal(boolean animated, boolean dynamicSafe) {
118                    this();
119                    setAnimation(animated);
120            setDynamicSafe(dynamicSafe);
121            }
122    
123        /**
124         * Setup the modal to prevent memory leaks.
125         * When modal is hidden, will remove all event handlers, and them
126         * remove the modal DOM from document DOM.
127         *
128         * Default is false.
129         *
130         * @param dynamicSafe
131         */
132        public void setDynamicSafe(boolean dynamicSafe) {
133            if (dynamicSafe) {
134                addHiddenHandler(new HiddenHandler() {
135    
136                    @Override
137                    public void onHidden(HiddenEvent hiddenEvent) {
138                        (new Timer() {
139                            @Override
140                            public void run() {
141                                unsetHandlerFunctions(getElement());
142                                RootPanel.get().remove(Modal.this);
143                            }
144                        }).schedule(3000);
145                    }
146                });
147            }
148        }
149    
150            /**
151             * Sets the title of the Modal.
152             * 
153             * @param title
154             *            the title of the Modal
155             */
156            @Override
157            public void setTitle(String title) {
158                    header.clear();
159                    if (title == null || title.isEmpty()) {
160                            showHeader(false);
161                    } else {
162                            header.add(close);
163                            header.add(new Heading(3, title));
164                            showHeader(true);
165                    }
166            }
167    
168            private void showHeader(boolean show) {
169                    if (show)
170                            header.setStyleName(Constants.MODAL_HEADER);
171                    else
172                            header.removeStyleName(Constants.MODAL_HEADER);
173            }
174    
175            /**
176             * {@inheritDoc}
177             */
178            public void setAnimation(boolean animated) {
179                    if (animated)
180                            addStyleName(Constants.FADE);
181                    else
182                            removeStyleName(Constants.FADE);
183            }
184    
185            /**
186             * {@inheritDoc}
187             */
188            public boolean getAnimation() {
189                    return getStyleName().contains(Constants.FADE);
190            }
191    
192            /**
193             * Sets whether this Modal appears on top of others or is the only one
194             * visible on screen.
195             * 
196             * @param hideOthers
197             *            <code>true</code> to make sure that this modal is the only one
198             *            shown. All others will be hidden. Default: <code>true</code>
199             */
200            public void setHideOthers(boolean hideOthers) {
201                    this.hideOthers = hideOthers;
202            }
203    
204            /**
205             * Sets whether the Modal is closed when the <code>ESC</code> is pressed.
206             * 
207             * @param keyboard
208             *            <code>true</code> if the Modal is closed by <code>ESC</code>
209             *            key. Default: <code>true</code>
210             */
211            public void setKeyboard(boolean keyboard) {
212                    this.keyboard = keyboard;
213                    reconfigure();
214            }
215    
216            /**
217             * Sets the type of the backdrop.
218             * 
219             * @param type
220             *            the backdrop type
221             */
222            public void setBackdrop(BackdropType type) {
223                    backdropType = type;
224                    reconfigure();
225            }
226    
227            /**
228             * Reconfigures the modal with changed settings.
229             */
230            protected void reconfigure() {
231                    if (configured) {
232                            configure(keyboard, backdropType, show);
233                    }
234            }
235    
236            /**
237             * {@inheritDoc}
238             */
239            @Override
240            public void add(Widget w) {
241                    if (w instanceof ModalFooter) {
242                            super.add(w);
243                    } else
244                            body.add(w);
245            }
246    
247            /**
248             * {@inheritDoc}
249             */
250            @Override
251            public void insert(Widget w, int beforeIndex) {
252                    body.insert(w, beforeIndex);
253            }
254    
255            /**
256             * {@inheritDoc}
257             */
258            @Override
259            protected void onLoad() {
260                    super.onLoad();
261                    configure(keyboard, backdropType, show);
262                    configured = true;
263            }
264    
265            /**
266             * {@inheritDoc}
267             */
268            public void show() {
269                    changeVisibility("show");
270            }
271    
272            /**
273             * {@inheritDoc}
274             */
275            public void hide() {
276                    changeVisibility("hide");
277            }
278    
279            /**
280             * {@inheritDoc}
281             */
282            public void toggle() {
283                    changeVisibility("toggle");
284            }
285    
286            private void changeVisibility(String visibility) {
287                    changeVisibility(getElement(), visibility);
288            }
289    
290            /**
291             * This method is called immediately when the widget's {@link #hide()}
292             * method is executed.
293             */
294            protected void onHide() {
295                    fireEvent(new HideEvent());
296            }
297    
298            /**
299             * This method is called once the widget is completely hidden.
300             */
301            protected void onHidden() {
302                    fireEvent(new HiddenEvent());
303                    currentlyShown.remove(this);
304            }
305    
306            /**
307             * This method is called immediately when the widget's {@link #show()}
308             * method is executed.
309             */
310            protected void onShow() {
311                    if (hideOthers)
312                            hideShownModals();
313                    fireEvent(new ShowEvent());
314            }
315    
316            private void hideShownModals() {
317                    for (Modal m : currentlyShown)
318                            m.hide();
319            }
320    
321            /**
322             * This method is called once the widget is completely shown.
323             */
324            protected void onShown() {
325                    fireEvent(new ShownEvent());
326                    currentlyShown.add(this);
327            }
328    
329            private void configure(boolean keyboard, BackdropType backdropType,
330                            boolean show) {
331                    if (backdropType == BackdropType.NORMAL)
332                            configure(getElement(), keyboard, true, show);
333                    else if (backdropType == BackdropType.NONE)
334                            configure(getElement(), keyboard, false, show);
335                    else if (backdropType == BackdropType.STATIC)
336                            configure(getElement(), keyboard, BackdropType.STATIC.get(), show);
337            }
338    
339            private native void configure(Element e, boolean k, boolean b, boolean s) /*-{
340                    $wnd.jQuery(e).modal({
341                            keyboard : k,
342                            backdrop : b,
343                            show : s
344                    });
345            }-*/;
346    
347            private native void configure(Element e, boolean k, String b, boolean s) /*-{
348                    $wnd.jQuery(e).modal({
349                            keyboard : k,
350                            backdrop : b,
351                            show : s
352                    });
353            }-*/;
354    
355            private native void changeVisibility(Element e, String visibility) /*-{
356                    $wnd.jQuery(e).modal(visibility);
357            }-*/;
358    
359            /**
360             * Links the Java functions that fire the events.
361             */
362            private native void setHandlerFunctions(Element e) /*-{
363                    var that = this;
364                    $wnd.jQuery(e).on('hide', function() {
365                            that.@com.github.gwtbootstrap.client.ui.Modal::onHide()();
366                    });
367                    $wnd.jQuery(e).on('hidden', function() {
368                            that.@com.github.gwtbootstrap.client.ui.Modal::onHidden()();
369                    });
370                    $wnd.jQuery(e).on('show', function() {
371                            that.@com.github.gwtbootstrap.client.ui.Modal::onShow()();
372                    });
373                    $wnd.jQuery(e).on('shown', function() {
374                            that.@com.github.gwtbootstrap.client.ui.Modal::onShown()();
375                    });
376            }-*/;
377    
378    
379        /**
380         * Unlinks all the Java functions that fire the events.
381         */
382        private native void unsetHandlerFunctions(Element e) /*-{
383            $wnd.jQuery(e).off();
384        }-*/;
385    
386            /**
387             * {@inheritDoc}
388             */
389            public HandlerRegistration addHideHandler(HideHandler handler) {
390                    return addHandler(handler, HideEvent.getType());
391            }
392    
393            /**
394             * {@inheritDoc}
395             */
396            public HandlerRegistration addHiddenHandler(HiddenHandler handler) {
397                    return addHandler(handler, HiddenEvent.getType());
398            }
399    
400            /**
401             * {@inheritDoc}
402             */
403            public HandlerRegistration addShowHandler(ShowHandler handler) {
404                    return addHandler(handler, ShowEvent.getType());
405            }
406    
407            /**
408             * {@inheritDoc}
409             */
410            public HandlerRegistration addShownHandler(ShownHandler handler) {
411                    return addHandler(handler, ShownEvent.getType());
412            }
413            
414            /**
415             * Show/Hide close button. The Modal must have a title.
416             * 
417             * @param visible <b>true</b> for show and <b>false</b> to hide. Defaults is <b>true</b>.
418             */
419            public void setCloseVisible(boolean visible) {
420                close.getElement().getStyle().setVisibility(visible ? 
421                        Style.Visibility.VISIBLE : Style.Visibility.HIDDEN);
422            }
