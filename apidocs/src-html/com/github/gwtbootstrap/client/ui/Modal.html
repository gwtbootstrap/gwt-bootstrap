016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.Close.DataDismiss;
019import com.github.gwtbootstrap.client.ui.base.DivWidget;
020import com.github.gwtbootstrap.client.ui.base.IsAnimated;
021import com.github.gwtbootstrap.client.ui.base.HasVisibleHandlers;
022import com.github.gwtbootstrap.client.ui.event.HiddenEvent;
023import com.github.gwtbootstrap.client.ui.event.HiddenHandler;
024import com.github.gwtbootstrap.client.ui.event.HideEvent;
025import com.github.gwtbootstrap.client.ui.event.HideHandler;
026import com.github.gwtbootstrap.client.ui.event.ShowEvent;
027import com.github.gwtbootstrap.client.ui.event.ShowHandler;
028import com.github.gwtbootstrap.client.ui.event.ShownEvent;
029import com.github.gwtbootstrap.client.ui.event.ShownHandler;
030import com.google.gwt.dom.client.Element;
031import com.google.gwt.event.shared.HandlerRegistration;
032import com.google.gwt.user.client.ui.RootPanel;
033import com.google.gwt.user.client.ui.Widget;
034
035/**
036 * 
037 * @author Carlos A Becker
038 * @author Dominik Mayer
039 * 
040 */
041public class Modal extends DivWidget implements HasVisibleHandlers,
042                IsAnimated {
043
044        private final DivWidget header = new DivWidget();
045        private final DivWidget body = new DivWidget("modal-body");
046
047        private boolean keyboard = true;
048        private boolean backdrop = true;
049        private boolean show = false;
050
051        private boolean configured = false;
052
053        public Modal() {
054                super("modal");
055                super.add(header);
056                super.add(body);
057                setVisible(false);
058                RootPanel.get().add(this);
059                setHandlerFunctions();
060        }
061
062        public Modal(boolean animated) {
063                this();
064                setAnimated(animated);
065        }
066
067        @Override
068        public void setTitle(String title) {
069                header.clear();
070                if (title == null || title.isEmpty()) {
071                        showHeader(false);
072                } else {
073                        header.add(new Close(DataDismiss.MODAL));
074                        header.add(new Heading(title, 3));
075                        showHeader(true);
076                }
077        }
078        
079        private void showHeader(boolean show) {
080                if (show)
081                        header.setStyleName("modal-header");
082                else
083                        header.removeStyleName("modal-header");
084        }
085
086        public void setAnimated(boolean animated) {
087                if (animated)
088                        addStyleName("fade");
089                else
090                        removeStyleName("fade");
091        }
092
093        public void setShow(boolean show) {
094                this.show = show;
095                reconfigure();
096        }
097
098        public void setKeyboard(boolean keyboard) {
099                this.keyboard = keyboard;
100                reconfigure();
101        }
102
103        public void setBackdrop(boolean backdrop) {
104                this.backdrop = backdrop;
105                reconfigure();
106        }
107
108        public void reconfigure() {
109                if (configured) {
110                        configure(getElement(), keyboard, backdrop, show);
111                }
112        }
113
114        @Override
115        public void add(Widget w) {
116                if(w instanceof ModalFooter) {
117                        super.add(w);
118                } else body.add(w);
119        }
120        
121        @Override
122        public void insert(Widget w, int beforeIndex) {
123                body.insert(w, beforeIndex);
124        }
125
126        @Override
127        protected void onLoad() {
128                super.onLoad();
129                configure(getElement(), keyboard, backdrop, show);
130                configured = true;
131        }
132        
133        /**
134         * {@inheritDoc}
135         */
136        public void show() {
137                changeVisibility("show");
138        }
139
140        /**
141         * {@inheritDoc}
142         */
143        public void hide() {
144                changeVisibility("hide");
145        }
146
147        /**
148         * {@inheritDoc}
149         */
150        public void toggle() {
151                changeVisibility("toggle");
152        }
153        
154        private void changeVisibility(String visibility) {
155                changeVisibility(getElement(), visibility);
156        }
157        
158        /**
159         * {@inheritDoc}
160         */
161        public HandlerRegistration addHideHandler(HideHandler handler) {
162                addHideHandler(getElement());
163                return addHandler(handler, HideEvent.getType());
164        }
165        
166        /**
167         * {@inheritDoc}
168         */
169        public HandlerRegistration addHiddenHandler(HiddenHandler handler) {
170                addHiddenHandler(getElement());
171                return addHandler(handler, HiddenEvent.getType());
172        }
173        
174        /**
175         * {@inheritDoc}
176         */
177        public HandlerRegistration addShowHandler(ShowHandler handler) {
178                addShowHandler(getElement());
179                return addHandler(handler, ShowEvent.getType());
180        }
181        
182        /**
183         * {@inheritDoc}
184         */
185        public HandlerRegistration addShownHandler(ShownHandler handler) {
186                addShownHandler(getElement());
187                return addHandler(handler, ShownEvent.getType());
188        }
189        
190        private void fireHideEvent() {
191                fireEvent(new HideEvent());
192        }
193        
194        private void fireHiddenEvent() {
195                fireEvent(new HiddenEvent());
196        }
197
198        private void fireShowEvent() {
199                fireEvent(new ShowEvent());
200        }
201
202        private void fireShownEvent() {
203                fireEvent(new ShownEvent());
204        }
205        
206        /*
207         * native functions
208         */
209        private native void configure(Element e, boolean k, boolean b, boolean s) /*-{
210                $wnd.jQuery(e).modal({
211                        keyboard : k,
212                        backdrop : b,
213                        show : s
214                });
215        }-*/;
216
217        private native void changeVisibility(Element e, String visibility) /*-{
218                $wnd.jQuery(e).modal(visibility);
219        }-*/;
220        
221        /**
222         * Adds the Java functions that fire the Events to document. It is a 
223         * convenience method to have a cleaner code later on.
224         */
225        private native void setHandlerFunctions() /*-{
226                var that = this;
227                $wnd.fireHideEvent = $entry(function() {
228          that.@com.github.gwtbootstrap.client.ui.Modal::fireHideEvent()();});
229                $wnd.fireHiddenEvent = $entry(function() {
230          that.@com.github.gwtbootstrap.client.ui.Modal::fireHiddenEvent()();});
231                $wnd.fireShowEvent = $entry(function() {
232          that.@com.github.gwtbootstrap.client.ui.Modal::fireShowEvent()();});
233                $wnd.fireShownEvent = $entry(function() {
234          that.@com.github.gwtbootstrap.client.ui.Modal::fireShownEvent()();});
235        }-*/;
236
237        private native void addHideHandler(Element e) /*-{
238                $wnd.jQuery(e).on('hide', function() {
239                        $wnd.fireHideEvent();
240                } );
241        }-*/;
242
243        private native void addHiddenHandler(Element e) /*-{
244                $wnd.jQuery(e).on('hidden', function() {
245                        $wnd.fireHiddenEvent();
246                } );
247        }-*/;
248        
249        private native void addShowHandler(Element e) /*-{
250                $wnd.jQuery(e).on('show', function() {
251                        $wnd.fireShowEvent();
252                } );
253        }-*/;
254        
255        private native void addShownHandler(Element e) /*-{
256                $wnd.jQuery(e).on('shown', function() {
257                        $wnd.fireShownEvent();
258                } );
259        }-*/;
260}
