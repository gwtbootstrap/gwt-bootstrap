001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.MarkupWidget;
004import com.github.gwtbootstrap.client.ui.constants.Constants;
005import com.google.gwt.user.client.Element;
006import com.google.gwt.user.client.ui.Widget;
007
008/**
009 * Markup widget of CollapseTrigger
010 * <p>
011 * It's a markup widget (decorator widget).
012 * it's can exchange child to {@link Collapse}'s toggle trigger widget.
013 * 
014 * <p>
015 * <h3>UiBinder Usage:</h3>
016 * </p>
017 * <pre>
018 * {@code
019 * <b:CollapseTrigger target="#toggle1">
020 *      <b:Button>Collapse Trigger</b:Button>
021 * </b:CollapseTrigger>
022 *
023 * <!-- if use collapse trigger, you must set existTrigger=true attribute to Collapse tag. -->
024 * <b:Collapse existTrigger="true" b:id="toggle1" defaultOpen="true">
025 *     <b:FluidRow>
026 *         <b:Column size="12">
027 *             <b:Alert close="false" animation="true" heading="collapsible1">
028 *                 <b:Paragraph>
029 *                     Hello :D
030 *                 </b:Paragraph>
031 *             </b:Alert>
032 *         </b:Column>
033 *     </b:FluidRow>
034 * </b:Collapse>
035 * }
036 * </pre>
037 * 
038 * @since 2.1.1.0
039 * @author ohashi keisuke
040 * @see Accordion
041 * @see Collapse
042 * @see CollapseTrigger
043 * @see <a href="http://twitter.github.com/bootstrap/javascript.html#collapse">Twitter Bootstrap document</a>
044 *
045 */
046public class CollapseTrigger extends MarkupWidget {
047
048    private String target;
049    
050    private String parent;
051    
052    /**
053     * Create an empty widget with target selector
054     * @param target selector (eg: #myCollapse)
055     */
056    public CollapseTrigger(String target) {
057        this.target = target;
058    }
059    
060    /**
061     * Create an empty widget.
062     */
063    public CollapseTrigger() {
064    }
065    
066    /**
067     * {@inheritDoc}
068     */
069    @Override
070    public Widget asWidget() {
071        
072        if(widget != null) {
073            Element element = widget.getElement();
074            element.setAttribute(Constants.DATA_TOGGLE, Constants.COLLAPSE);
075            element.setAttribute(Constants.DATA_TARGET, target);
076            
077            if(parent != null && !parent.isEmpty()) {
078                setParent(parent);
079            }
080            
081        }
082        
083        return super.asWidget();
084    }
085    
086    /**
087     * Set target collapse selector.
088     * @param target selector of target. (eg:#myCollapse)
089     */
090    public void setTarget(String target) {
091        this.target = target;
092        
093        if(widget != null) {
094            Element element = widget.getElement();
095            element.setAttribute(Constants.DATA_TARGET, target);
096        }
097    }
098    
099    /**
100     * Get target collapse selector.
101     * @return selector.
102     */
103    public String getTarget(){
104        return this.target;
105    }
106
107
108    /**
109     * Set parent selector.
110     * 
111     * it only work with {@link AccordionGroup},
112     * Please see <a href="https://github.com/twitter/bootstrap/issues/4988">this issue</a>.
113     * 
114     * @param parent parent selector
115     */
116    public void setParent(String parent) {
117        this.parent = parent;
118        
119        if(widget != null) {
120            widget.getElement().setAttribute("data-parent", parent);
121        }
122        
123    }
124    
125}
