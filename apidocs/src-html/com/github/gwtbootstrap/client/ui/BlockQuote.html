016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.base.AbstractTypography;
019import com.google.gwt.user.client.DOM;
020
021/**
022 *
023 * @author carlos
024 */
025public class BlockQuote extends AbstractTypography {
026
027    {
028        setElement(DOM.createElement("blockquote"));
029    }
030
031    public BlockQuote() {
032        // Needed for UiBinder
033    }
034    
035    public BlockQuote(String text) {
036        setText(text);
037    }
038
039    public BlockQuote(String text, String cite) {
040        this(text);
041        setCite(cite);
042    }
043
044    public BlockQuote(String text, boolean pullright) {
045        setText(text);
046        setPullright(pullright);
047    }
048
049    public BlockQuote(String text, String cite, boolean pullright) {
050        this(text);
051        setCite(cite);
052        setPullright(pullright);
053    }
054
055    private void pullRight() {
056        setStyleName("pull-right");
057    }
058    
059    public void setCite(String cite) {
060        getElement().appendChild(new SmallCite(cite).getElement());
061    }
062    
063    public void setPullright(boolean pullright) {
064        if (pullright) {
065            pullRight();
066        }
067    }
068
069    private class Cite extends AbstractTypography {
070
071        {
072            setElement(DOM.createElement("cite"));
073        }
074
075        public Cite(String text) {
076            setText(text);
077        }
078    }
079
080    private class SmallCite extends AbstractTypography {
081
082        private final Cite cite;
083
084        {
085            setElement(DOM.createElement("small"));
086        }
087
088        public SmallCite(String text) {
089            this.cite = new Cite(text);
090            getElement().appendChild(cite.getElement());
091        }
092
093        @Override
094        public void setText(String text) {
095            cite.setText(text);
096        }
097    }
098}
