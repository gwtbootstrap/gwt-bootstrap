016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.base.StyleHelper;
019import com.google.gwt.core.client.GWT;
020import com.google.gwt.resources.client.CssResource.ImportedWithPrefix;
021import com.google.gwt.resources.client.ImageResource;
022import com.google.gwt.user.client.ui.Image;
023import com.google.gwt.user.client.ui.Widget;
024import com.google.gwt.view.client.ProvidesKey;
025
026/**
027 * 
028 * CellTable for Bootstrap style.
029 * 
030 * @since 2.0.4.0
031 * @author ohashi keisuke
032 *
033 * @param <T> Data Set
034 */
035public class CellTable<T> extends com.google.gwt.user.cellview.client.CellTable<T> {
036
037        public static enum TableType implements com.github.gwtbootstrap.client.ui.base.Style {
038                BORDERED("table-bordered"),
039                STRIPED("table-striped"),
040                CONDENSED("table-condensed");
041                
042                private final String styleName;
043
044                private TableType(String styleName) {
045                        this.styleName = styleName;
046                }
047
048                @Override
049                public String get() {
050                        return styleName;
051                }
052        }
053        
054        /**
055         * The default page size.
056         */
057        private static final int DEFAULT_PAGESIZE = 15;
058
059        private static Resources DEFAULT_RESOURCES = null;
060
061        private static Resources getDefaultResources() {
062                if (DEFAULT_RESOURCES == null) {
063                        DEFAULT_RESOURCES = GWT.create(Resources.class);
064                }
065                return DEFAULT_RESOURCES;
066        }
067
068        public interface Resources extends com.google.gwt.user.cellview.client.CellTable.Resources {
069
070                @Override
071                @Source(Style.DEFAULT_CSS)
072                public Style cellTableStyle();
073        }
074
075        @ImportedWithPrefix("gwt-bootstrap-cellTable")
076        public interface Style extends com.google.gwt.user.cellview.client.CellTable.Style {
077
078                String DEFAULT_CSS = "com/github/gwtbootstrap/client/ui/GwtBootstrapCellTable.css";
079        }
080
081        public CellTable() {
082                this(DEFAULT_PAGESIZE);
083        }
084
085        public CellTable(int pageSize, ProvidesKey<T> keyProvider) {
086                this(pageSize, keyProvider , createDefaultLoadingIndicator(getDefaultResources()));
087        }
088
089        public CellTable(int pageSize, ProvidesKey<T> keyProvider, Widget loadingIndicator) {
090                super(pageSize, getDefaultResources(), keyProvider, loadingIndicator);
091                setStyleName("table");
092        }
093
094        public CellTable(int pageSize) {
095                this(pageSize, null);
096        }
097
098        public CellTable(ProvidesKey<T> keyProvider) {
099                this(DEFAULT_PAGESIZE, keyProvider);
100        }
101        
102        public void setStriped(boolean striped) {
103                if(striped) {
104                        StyleHelper.addStyle(this, TableType.STRIPED);
105                } else {
106                        StyleHelper.removeStyle(this, TableType.STRIPED);
107                }
108        }
109        
110        public void setBordered(boolean bordered) {
111                if(bordered) {
112                        StyleHelper.addStyle(this, TableType.BORDERED);
113                } else {
114                        StyleHelper.removeStyle(this, TableType.BORDERED);
115                }
116        }
117
118        public void setCondensed(boolean condensed) {
119                if(condensed) {
120                        StyleHelper.addStyle(this, TableType.CONDENSED);
121                } else {
122                        StyleHelper.removeStyle(this, TableType.CONDENSED);
123                }
124        }
125        /**
126         * Create the default loading indicator using the loading image in the
127         * specified {@link Resources}.
128         * 
129         * @param resources
130         *            the resources
131         * @return a widget loading indicator
132         */
133        private static Widget createDefaultLoadingIndicator(Resources resources) {
134                ImageResource loadingImg = resources.cellTableLoading();
135                return (loadingImg == null)
136                                                                        ? null
137                                                                        : new Image(loadingImg);
138        }
139}
