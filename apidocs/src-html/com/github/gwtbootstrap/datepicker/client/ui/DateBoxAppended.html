001package com.github.gwtbootstrap.datepicker.client.ui;
002
003import com.github.gwtbootstrap.client.ui.AddOn;
004import com.github.gwtbootstrap.client.ui.AppendButton;
005import com.github.gwtbootstrap.client.ui.constants.IconType;
006import com.github.gwtbootstrap.datepicker.client.ui.base.DateBoxBase;
007import com.github.gwtbootstrap.datepicker.client.ui.base.HasDateFormat;
008import com.google.gwt.event.logical.shared.ValueChangeHandler;
009import com.google.gwt.event.shared.HandlerRegistration;
010import com.google.gwt.user.client.ui.HasValue;
011
012import java.util.Date;
013
014/**
015 * DateBoxAppended component.
016 *
017 * @since 2.0.3.0
018 * @author Carlos Alexandro Becker
019 */
020public class DateBoxAppended extends AppendButton implements HasValue<Date>, HasDateFormat {
021    private final DateBoxAppendedAdapter box;
022
023    {
024        this.box = new DateBoxAppendedAdapter();
025        AddOn icon = new AddOn();
026        icon.setIcon(IconType.TH);
027        add(box);
028        add(icon);
029    }
030
031    /**
032     * {@inheritDoc}
033     */
034    @Override
035    public Date getValue() {
036        return box.getValue();
037    }
038
039    /**
040     * {@inheritDoc}
041     */
042    @Override
043    public void setValue(Date value) {
044        box.setValue(value);
045    }
046
047    /**
048     * {@inheritDoc}
049     */
050    @Override
051    public void setValue(Date value, boolean fireEvents) {
052        box.setValue(value, fireEvents);
053    }
054
055    /**
056     * {@inheritDoc}
057     */
058    @Override
059    public HandlerRegistration addValueChangeHandler(ValueChangeHandler<Date> dateValueChangeHandler) {
060        return box.addValueChangeHandler(dateValueChangeHandler);
061    }
062
063    /**
064     * {@inheritDoc}
065     */
066    @Override
067    public void setFormat(String format) {
068        box.setFormat(format);
069    }
070
071    /**
072     * An 'adapter' to change some aspects of the behavior of datepickerbase. //TODO may need refactor.
073     */
074    private class DateBoxAppendedAdapter extends DateBoxBase {
075        @Override
076        protected void configure() {
077            DateBoxAppended that = DateBoxAppended.this;
078            super.configure(that);
079            getBox().getElement().setAttribute("readonly", null);
080            that.addStyleName("date");
081        }
082    }
083}
