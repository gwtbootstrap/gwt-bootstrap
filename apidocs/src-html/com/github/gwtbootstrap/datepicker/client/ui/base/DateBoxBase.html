060/**

063 * @author Carlos Alexandro Becker

067public class DateBoxBase extends Widget implements HasValue<Date>, HasValueChangeHandlers<Date>, HasVisibility,
068        HasChangeHandlers, HasVisibleHandlers, HasAllDatePickerHandlers, IsEditor<TakesValueEditor<Date>>, HasPlaceholder, HasAlternateSize, IsSearchQuery, HasSize, HasId, IsResponsive , HasStyle {

070    private final TextBox box;
071    private String format;
072    private String language;
073    private DateTimeFormat dtf;
074    private TakesValueEditor<Date> editor;
075
076    /** placeholderHelper */
077    private PlaceholderHelper placeholderHelper = GWT.create(PlaceholderHelper.class);
078    private boolean autoclose = false;
079
080    public DateBoxBase() {
081        this.box = new TextBox();
082        setElement(box.getElement());
083        setFormat("mm/dd/yyyy");
084        this.language = LocaleUtil.getLanguage();
085        setValue(new Date());
086    }
087
088    /**
089     * {@inheritDoc}
090     */
091    @Override
092    public void setFormat(String format) {
093        this.format = format;
094        Date oldValue = getValue();
095        this.dtf = DateTimeFormat.getFormat(format.replaceAll("mm", "MM"));
096        if (oldValue != null) {
097            setValue(oldValue);
098        }
099    }
100
101    public void setLanguage(String language) {
102        this.language = language;
103        LocaleUtil.forceLocale(language);
104    }
105
106    /**
107     * Returns the internal instance of textbox element. Use only if know what you are doing.
108     *
109     * @return internal textbox intance.
110     */
111    protected TextBox getBox() {
112        return box;
113    }
114
115    /**
116     * {@inheritDoc}
117     */
118    @Override
119    public Date getValue() {
120        try {
121            return dtf != null && box.getValue() != null ? dtf.parse(box.getValue()) : null;
122        } catch(Exception e) {
123            return null;
124        }
125    }
126    
127    /**
128     * Get un-tranceform text
129     * @return text box value
130     */
131    public String getOriginalValue() {
132        return box.getValue();
133    }
134
135    /**
136     * {@inheritDoc}
137     */
138    @Override
139    public void setValue(Date value) {
140        setValue(value, false);
141    }
142
143    /**
144     * {@inheritDoc}
145     */
146    @Override
147    public void setValue(Date value, boolean fireEvents) {
148        box.setValue(value != null ? dtf.format(value) : null);
149        
150        updateValue(box.getElement());
151        
152        if (fireEvents) {
153            ValueChangeEvent.fire(this, value);
154        }
155    }
156    
157    protected native void updateValue(Element e)/*-{
158        if($wnd.jQuery(e).data('datepicker')) {
159            $wnd.jQuery(e).data('datepicker').update();
160        }
161    }-*/;

180    /**
181     * Configure the elements for a specific widget.
182     * Use only if you know what you are doing.
183     *
184     * @param w: the widget to configure.
185     */
186    protected void configure(Widget w) {
187        w.getElement().setAttribute("data-date-format", format);
188        w.getElement().setAttribute("data-date-language", language);
189        configure(w.getElement(), autoclose);
190    }

193     * dateChange event handler.

197    }

219        var $that = $wnd.jQuery(e);
220        if($that.data('datepicker')) {
221            console.log($that.data());
222            $that.removeData('dateFormat');
223            $that.removeData('dateLanguage');
224            $that.removeData('dateWeekstart');
225            $that.removeData('dateStartdate');
226            $that.removeData('dateEnddate');

228            $that.off();
229            console.log($that.data());
230        }
231    }-*/;

241        $wnd.jQuery(e).datepicker({autoclose : autoclose})

253    private native void execute(Element e, String cmd) /*-{
254        $wnd.jQuery(e).datepicker(cmd);

259    }
